57 std::string AssertionResult::getExpression() const {
58 // Possibly overallocating by 3 characters should be basically free
59 std::string expr; expr.reserve(m_info.capturedExpression.size() + 3);
60 if (isFalseTest(m_info.resultDisposition)) {
61 expr += "!(";
62 }
63 expr += m_info.capturedExpression;
64 if (isFalseTest(m_info.resultDisposition)) {
65 expr += ')';
66 }
67 return expr;
68 }

70 std::string AssertionResult::getExpressionInMacro() const {
