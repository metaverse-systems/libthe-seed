20TEST_CASE("unique_ptr reimplementation: basic functionality", "[internals][unique-ptr]") {
21 using Catch::Detail::unique_ptr;
22 SECTION("Default constructed unique_ptr is empty") {
23 unique_ptr<int> ptr;
24 REQUIRE_FALSE(ptr);
25 REQUIRE(ptr.get() == nullptr);
26 }
27 SECTION("Take ownership of allocation") {
28 auto naked_ptr = new int{ 0 };
29 unique_ptr<int> ptr(naked_ptr);
30 REQUIRE(ptr);
31 REQUIRE(*ptr == 0);
32 REQUIRE(ptr.get() == naked_ptr);
33 SECTION("Plain reset deallocates") {
34 ptr.reset(); // this makes naked_ptr dangling!
35 REQUIRE_FALSE(ptr);
36 REQUIRE(ptr.get() == nullptr);
37 }
38 SECTION("Reset replaces ownership") {
39 ptr.reset(new int{ 2 });
40 REQUIRE(ptr);
41 REQUIRE(ptr.get() != nullptr);
42 REQUIRE(*ptr == 2);
43 }
44 }
45 SECTION("Release releases ownership") {
46 auto naked_ptr = new int{ 1 };
47 unique_ptr<int> ptr(naked_ptr);
48 ptr.release();
49 CHECK_FALSE(ptr);
50 CHECK(ptr.get() == nullptr);
51 delete naked_ptr;
52 }
53 SECTION("Move constructor") {
54 unique_ptr<int> ptr1(new int{ 1 });
55 auto ptr2(std::move(ptr1));
56 REQUIRE_FALSE(ptr1);
57 REQUIRE(ptr2);
58 REQUIRE(*ptr2 == 1);
59 }
60 SECTION("Move assignment") {
61 unique_ptr<int> ptr1(new int{ 1 }), ptr2(new int{ 2 });
62 ptr1 = std::move(ptr2);
63 REQUIRE_FALSE(ptr2);
64 REQUIRE(ptr1);
65 REQUIRE(*ptr1 == 2);
66 }
67 SECTION("free swap") {
68 unique_ptr<int> ptr1(new int{ 1 }), ptr2(new int{ 2 });
69 swap(ptr1, ptr2);
70 REQUIRE(*ptr1 == 2);
71 REQUIRE(*ptr2 == 1);
72 }
73}
