46 IEventListenerPtr prepareReporters(Config const* config) {
47 if (Catch::getRegistryHub().getReporterRegistry().getListeners().empty()
48 && config->getProcessedReporterSpecs().size() == 1) {
49 auto const& spec = config->getProcessedReporterSpecs()[0];
50 return createReporter(
51 spec.name,
52 ReporterConfig( config,
53 makeStream( spec.outputFilename ),
54 spec.colourMode,
55 spec.customOptions ) );
56 }
57
58 auto multi = Detail::make_unique<MultiReporter>(config);
59
60 auto const& listeners = Catch::getRegistryHub().getReporterRegistry().getListeners();
61 for (auto const& listener : listeners) {
62 multi->addListener(listener->create(config));
63 }
64
65 for ( auto const& reporterSpec : config->getProcessedReporterSpecs() ) {
66 multi->addReporter( createReporter(
67 reporterSpec.name,
68 ReporterConfig( config,
69 makeStream( reporterSpec.outputFilename ),
70 reporterSpec.colourMode,
71 reporterSpec.customOptions ) ) );
72 }
73
74 return multi;
75 }

77 class TestGroup {
78 public:
79 explicit TestGroup(IEventListenerPtr&& reporter, Config const* config):
80 m_reporter(reporter.get()),
81 m_config{config},
82 m_context{config, CATCH_MOVE(reporter)} {
83
84 assert( m_config->testSpec().getInvalidSpecs().empty() &&
85 "Invalid test specs should be handled before running tests" );
86
87 auto const& allTestCases = getAllTestCasesSorted(*m_config);
88 auto const& testSpec = m_config->testSpec();
89 if ( !testSpec.hasFilters() ) {
90 for ( auto const& test : allTestCases ) {
91 if ( !test.getTestCaseInfo().isHidden() ) {
92 m_tests.emplace( &test );
93 }
94 }
95 } else {
96 m_matches =
97 testSpec.matchesByFilter( allTestCases, *m_config );
98 for ( auto const& match : m_matches ) {
99 m_tests.insert( match.tests.begin(),
100 match.tests.end() );
101 }
102 }
103
104 m_tests = createShard(m_tests, m_config->shardCount(), m_config->shardIndex());
105 }
106
107 Totals execute() {
108 Totals totals;
109 for (auto const& testCase : m_tests) {
110 if (!m_context.aborting())
111 totals += m_context.runTest(*testCase);
112 else
113 m_reporter->skipTest(testCase->getTestCaseInfo());
114 }
115
116 for (auto const& match : m_matches) {
117 if (match.tests.empty()) {
118 m_unmatchedTestSpecs = true;
119 m_reporter->noMatchingTestCases( match.name );
120 }
121 }
122
123 return totals;
124 }
125
126 bool hadUnmatchedTestSpecs() const {
127 return m_unmatchedTestSpecs;
128 }
129
130
131 private:
132 IEventListener* m_reporter;
133 Config const* m_config;
134 RunContext m_context;
135 std::set<TestCaseHandle const*> m_tests;
136 TestSpec::Matches m_matches;
137 bool m_unmatchedTestSpecs = false;
138 };

200 int Session::applyCommandLine( int argc, char const * const * argv ) {
201 if( m_startupExceptions )
202 return 1;
203
204 auto result = m_cli.parse( Clara::Args( argc, argv ) );
205
206 if( !result ) {
207 config();
208 getCurrentMutableContext().setConfig(m_config.get());
209 auto errStream = makeStream( "%stderr" );
210 auto colour = makeColourImpl( ColourMode::PlatformDefault, errStream.get() );
211
212 errStream->stream()
213 << colour->guardColour( Colour::Red )
214 << "\nError(s) in input:\n"
215 << TextFlow::Column( result.errorMessage() ).indent( 2 )
216 << "\n\n";
217 errStream->stream() << "Run with -? for usage\n\n" << std::flush;
218 return MaxExitCode;
219 }
220
221 if( m_configData.showHelp )
222 showHelp();
223 if( m_configData.libIdentify )
224 libIdentify();
225
226 m_config.reset();
227 return 0;
228 }

287 int Session::runInternal() {
288 if( m_startupExceptions )
289 return 1;
290
291 if (m_configData.showHelp || m_configData.libIdentify) {
292 return 0;
293 }
294
295 if ( m_configData.shardIndex >= m_configData.shardCount ) {
296 Catch::cerr() << "The shard count (" << m_configData.shardCount
297 << ") must be greater than the shard index ("
298 << m_configData.shardIndex << ")\n"
299 << std::flush;
300 return 1;
301 }
302
303 CATCH_TRY {
304 config(); // Force config to be constructed
305
306 seedRng( *m_config );
307
308 if (m_configData.filenamesAsTags) {
309 applyFilenamesAsTags();
310 }
311
312 // Set up global config instance before we start calling into other functions
313 getCurrentMutableContext().setConfig(m_config.get());
314
315 // Create reporter(s) so we can route listings through them
316 auto reporter = prepareReporters(m_config.get());
317
318 auto const& invalidSpecs = m_config->testSpec().getInvalidSpecs();
319 if ( !invalidSpecs.empty() ) {
320 for ( auto const& spec : invalidSpecs ) {
321 reporter->reportInvalidTestSpec( spec );
322 }
323 return 1;
324 }
325
326
327 // Handle list request
328 if (list(*reporter, *m_config)) {
329 return 0;
330 }
331
332 TestGroup tests { CATCH_MOVE(reporter), m_config.get() };
333 auto const totals = tests.execute();
334
335 if ( tests.hadUnmatchedTestSpecs()
336 && m_config->warnAboutUnmatchedTestSpecs() ) {
337 return 3;
338 }
339
340 if ( totals.testCases.total() == 0
341 && !m_config->zeroTestsCountAsSuccess() ) {
342 return 2;
343 }
344
345 if ( totals.testCases.total() > 0 &&
346 totals.testCases.total() == totals.testCases.skipped
347 && !m_config->zeroTestsCountAsSuccess() ) {
348 return 4;
349 }
350
351 // Note that on unices only the lower 8 bits are usually used, clamping
352 // the return value to 255 prevents false negative when some multiple
353 // of 256 tests has failed
354 return (std::min) (MaxExitCode, static_cast<int>(totals.assertions.failed));
355 }
356#if !defined(CATCH_CONFIG_DISABLE_EXCEPTIONS)
357 catch( std::exception& ex ) {
358 Catch::cerr() << ex.what() << '\n' << std::flush;
359 return MaxExitCode;
360 }
361#endif
362 }
