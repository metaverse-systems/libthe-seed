95 template<typename T>
96 class FixedValuesGenerator final : public IGenerator<T> {
97 static_assert(!std::is_same<T, bool>::value,

99 "specialization, use SingleValue Generator instead.");

102 public:

105 T const& get() const override {
106 return m_values[m_idx];
107 }

125 template<typename T>
126 class Generators : public IGenerator<T> {
127 std::vector<GeneratorWrapper<T>> m_generators;
128 size_t m_current = 0;
129
130 void add_generator( GeneratorWrapper<T>&& generator ) {
131 m_generators.emplace_back( CATCH_MOVE( generator ) );
132 }
133 void add_generator( T const& val ) {
134 m_generators.emplace_back( value( val ) );
135 }
136 void add_generator( T&& val ) {
137 m_generators.emplace_back( value( CATCH_MOVE( val ) ) );
138 }
139 template <typename U>
140 std::enable_if_t<!std::is_same<std::decay_t<U>, T>::value>
141 add_generator( U&& val ) {
142 add_generator( T( CATCH_FORWARD( val ) ) );
143 }
144
145 template <typename U> void add_generators( U&& valueOrGenerator ) {
146 add_generator( CATCH_FORWARD( valueOrGenerator ) );
147 }
148
149 template <typename U, typename... Gs>
150 void add_generators( U&& valueOrGenerator, Gs&&... moreGenerators ) {
151 add_generator( CATCH_FORWARD( valueOrGenerator ) );
152 add_generators( CATCH_FORWARD( moreGenerators )... );
153 }
154
155 public:
156 template <typename... Gs>
157 Generators(Gs &&... moreGenerators) {
158 m_generators.reserve(sizeof...(Gs));
159 add_generators(CATCH_FORWARD(moreGenerators)...);
160 }
161
162 T const& get() const override {
163 return m_generators[m_current].get();
164 }
165
166 bool next() override {
167 if (m_current >= m_generators.size()) {
168 return false;
169 }
170 const bool current_status = m_generators[m_current].next();
171 if (!current_status) {
172 ++m_current;
173 }
174 return m_current < m_generators.size();
175 }
176 };

212 template<typename L>
213 auto generate( StringRef generatorName, SourceLineInfo const& lineInfo, L const& generatorExpression ) -> typename decltype(generatorExpression())::type {
214 using UnderlyingType = typename decltype(generatorExpression())::type;
215
216 IGeneratorTracker* tracker = acquireGeneratorTracker( generatorName, lineInfo );
217 // Creation of tracker is delayed after generator creation, so
218 // that constructing generator can fail without breaking everything.
219 if (!tracker) {
220 tracker = createGeneratorTracker(
221 generatorName,
222 lineInfo,
223 Catch::Detail::make_unique<Generators<UnderlyingType>>(
224 generatorExpression() ) );
225 }
226
227 auto const& generator = static_cast<IGenerator<UnderlyingType> const&>( *tracker->getGenerator() );
228 return generator.get();
229 }
