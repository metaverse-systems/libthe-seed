64TEST_CASE("convertIntoString stringification helper", "[toString][approvals]") {
65 using namespace std::string_literals;
66 using Catch::Detail::convertIntoString;
67 using namespace Catch;
68
69 SECTION("No escaping") {
70 CHECK(convertIntoString(""_sr, false) == R"("")"s);
71 CHECK(convertIntoString("abcd"_sr, false) == R"("abcd")"s);
72 CHECK(convertIntoString("ab\ncd"_sr, false) == "\"ab\ncd\""s);
73 CHECK(convertIntoString("ab\r\ncd"_sr, false) == "\"ab\r\ncd\""s);
74 CHECK(convertIntoString("ab\"cd"_sr, false) == R"("ab"cd")"s);
75 }
76 SECTION("Escaping invisibles") {
77 CHECK(convertIntoString(""_sr, true) == R"("")"s);

79 CHECK(convertIntoString("ab\r\ncd"_sr, true) == R"("ab\r\ncd")"s);
80 CHECK(convertIntoString("ab\tcd"_sr, true) == R"("ab\tcd")"s);
81 CHECK(convertIntoString("ab\fcd"_sr, true) == R"("ab\fcd")"s);
