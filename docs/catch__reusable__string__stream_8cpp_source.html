18 // This class encapsulates the idea of a pool of ostringstreams that can be reused.
19 struct StringStreams {
20 std::vector<Detail::unique_ptr<std::ostringstream>> m_streams;
21 std::vector<std::size_t> m_unused;
22 std::ostringstream m_referenceStream; // Used for copy state/ flags from
23
24 auto add() -> std::size_t {
25 if( m_unused.empty() ) {
26 m_streams.push_back( Detail::make_unique<std::ostringstream>() );
27 return m_streams.size()-1;
28 }
29 else {
30 auto index = m_unused.back();
31 m_unused.pop_back();
32 return index;
33 }
34 }
35
36 void release( std::size_t index ) {
37 m_streams[index]->copyfmt( m_referenceStream ); // Restore initial flags and other state
38 m_unused.push_back(index);
39 }
40 };
