26 std::string TagAliasRegistry::expandAliases( std::string const& unexpandedTestSpec ) const {
27 std::string expandedTestSpec = unexpandedTestSpec;
28 for( auto const& registryKvp : m_registry ) {
29 std::size_t pos = expandedTestSpec.find( registryKvp.first );
30 if( pos != std::string::npos ) {
31 expandedTestSpec = expandedTestSpec.substr( 0, pos ) +
32 registryKvp.second.tag +
33 expandedTestSpec.substr( pos + registryKvp.first.size() );
34 }
35 }
36 return expandedTestSpec;
37 }

39 void TagAliasRegistry::add( std::string const& alias, std::string const& tag, SourceLineInfo const& lineInfo ) {
40 CATCH_ENFORCE( startsWith(alias, "[@") && endsWith(alias, ']'),
41 "error: tag alias, '" << alias << "' is not of the form [@alias name].\n" << lineInfo );
42
43 CATCH_ENFORCE( m_registry.insert(std::make_pair(alias, TagAlias(tag, lineInfo))).second,
44 "error: tag alias, '" << alias << "' already registered.\n"
45 << "\tFirst seen at: " << find(alias)->lineInfo << "\n"
46 << "\tRedefined at: " << lineInfo );
47 }
