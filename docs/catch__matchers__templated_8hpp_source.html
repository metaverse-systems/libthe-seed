37 template<std::size_t N, std::size_t M>
38 std::array<void const*, N + M> array_cat(std::array<void const*, N> && lhs, std::array<void const*, M> && rhs) {
39 std::array<void const*, N + M> arr{};
40 std::copy_n(lhs.begin(), N, arr.begin());
41 std::copy_n(rhs.begin(), M, arr.begin() + N);
42 return arr;
43 }
