103// Based on example from https://docs.cucumber.io/gherkin/reference/#scenario-outline
104// (thanks to https://github.com/catchorg/Catch2/issues/850#issuecomment-399504851)
105
106// Note that GIVEN, WHEN, and THEN now forward onto DYNAMIC_SECTION instead of SECTION.
107// DYNAMIC_SECTION takes its name as a stringstream-style expression, so can be formatted using
108// variables in scope - such as the generated variables here. This reads quite nicely in the
109// test name output (the full scenario description).
110
111static auto eatCucumbers( int start, int eat ) -> int { return start-eat; }
112
113SCENARIO("Eating cucumbers", "[generators][approvals]") {
114 using tuple_type = std::tuple<int, int, int>;
115 auto [start, eat, left] = GENERATE( table<int, int, int>(
116 { tuple_type{ 12, 5, 7 }, tuple_type{ 20, 5, 15 } } ) );
117
118 GIVEN( "there are " << start << " cucumbers" )
119 WHEN( "I eat " << eat << " cucumbers" )
120 THEN( "I should have " << left << " cucumbers" ) {
121 REQUIRE( eatCucumbers( start, eat ) == left );
122 }
123}
124#endif

127TEST_CASE("Generators -- adapters", "[generators][generic]") {
128 // TODO: This won't work yet, introduce GENERATE_VAR?
129 //auto numbers = Catch::Generators::values({ 1, 2, 3, 4, 5, 6 });
130 SECTION("Filtering by predicate") {
131 SECTION("Basic usage") {
132 // This filters out all odd (false) numbers, giving [2, 4, 6]
133 auto i = GENERATE(filter([] (int val) { return val % 2 == 0; }, values({ 1, 2, 3, 4, 5, 6 })));
134 REQUIRE(i % 2 == 0);
135 }
136 SECTION("Throws if there are no matching values") {
137 using namespace Catch::Generators;
138 REQUIRE_THROWS_AS(filter([] (int) {return false; }, value(1)), Catch::GeneratorException);
139 }
140 }
141 SECTION("Shortening a range") {
142 // This takes the first 3 elements from the values, giving back [1, 2, 3]
143 auto i = GENERATE(take(3, values({ 1, 2, 3, 4, 5, 6 })));
144 REQUIRE(i < 4);
145 }
146 SECTION("Transforming elements") {
147 SECTION("Same type") {
148 // This doubles values [1, 2, 3] into [2, 4, 6]
149 auto i = GENERATE(map([] (int val) { return val * 2; }, values({ 1, 2, 3 })));
150 REQUIRE(i % 2 == 0);
151 }
152 SECTION("Different type") {
153 // This takes a generator that returns ints and maps them into strings
154 auto i = GENERATE(map<std::string>([] (int val) { return std::to_string(val); }, values({ 1, 2, 3 })));
155 REQUIRE(i.size() == 1);
156 }
157 SECTION("Different deduced type") {
158 // This takes a generator that returns ints and maps them into strings
159 auto i = GENERATE(map([] (int val) { return std::to_string(val); }, values({ 1, 2, 3 })));
160 REQUIRE(i.size() == 1);
161 }
162 }
163 SECTION("Repeating a generator") {
164 // This will return values [1, 2, 3, 1, 2, 3]
165 auto j = GENERATE(repeat(2, values({ 1, 2, 3 })));
166 REQUIRE(j > 0);
167 }
168 SECTION("Chunking a generator into sized pieces") {
169 SECTION("Number of elements in source is divisible by chunk size") {
170 auto chunk2 = GENERATE(chunk(2, values({ 1, 1, 2, 2, 3, 3 })));
171 REQUIRE(chunk2.size() == 2);
172 REQUIRE(chunk2.front() == chunk2.back());
173 }
174 SECTION("Number of elements in source is not divisible by chunk size") {
175 auto chunk2 = GENERATE(chunk(2, values({ 1, 1, 2, 2, 3 })));
176 REQUIRE(chunk2.size() == 2);
177 REQUIRE(chunk2.front() == chunk2.back());
178 REQUIRE(chunk2.front() < 3);
179 }
180 SECTION("Chunk size of zero") {
181 auto chunk2 = GENERATE(take(3, chunk(0, value(1))));
182 REQUIRE(chunk2.size() == 0);
183 }
184 SECTION("Throws on too small generators") {
185 using namespace Catch::Generators;
186 REQUIRE_THROWS_AS(chunk(2, value(1)), Catch::GeneratorException);
187 }

243TEST_CASE("Copy and then generate a range", "[generators]") {
244 SECTION("from var and iterators") {
245 static auto data = make_data();
246
247 // It is important to notice that a generator is only initialized
248 // **once** per run. What this means is that modifying data will not
249 // modify the underlying generator.
250 auto elem = GENERATE_REF(from_range(data.begin(), data.end()));
251 REQUIRE(elem % 2 == 1);
252 }
253 SECTION("From a temporary container") {
254 auto elem = GENERATE(from_range(make_data_counted()));
255 ++test_count;
256 REQUIRE(elem % 2 == 1);
257 }
258 SECTION("Final validation") {
259 REQUIRE(call_count == 1);
