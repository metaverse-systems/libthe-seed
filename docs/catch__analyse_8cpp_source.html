22 SampleAnalysis analyse(const IConfig &cfg, FDuration* first, FDuration* last) {
23 if (!cfg.benchmarkNoAnalysis()) {
24 std::vector<double> samples;
25 samples.reserve(static_cast<size_t>(last - first));
26 for (auto current = first; current != last; ++current) {
27 samples.push_back( current->count() );
28 }
29
30 auto analysis = Catch::Benchmark::Detail::analyse_samples(
31 cfg.benchmarkConfidenceInterval(),
32 cfg.benchmarkResamples(),
33 samples.data(),
34 samples.data() + samples.size() );
35 auto outliers = Catch::Benchmark::Detail::classify_outliers(
36 samples.data(), samples.data() + samples.size() );
37
38 auto wrap_estimate = [](Estimate<double> e) {
39 return Estimate<FDuration> {
40 FDuration(e.point),
41 FDuration(e.lower_bound),
42 FDuration(e.upper_bound),
43 e.confidence_interval,
44 };
45 };
46 std::vector<FDuration> samples2;
47 samples2.reserve(samples.size());
48 for (auto s : samples) {
49 samples2.push_back( FDuration( s ) );
50 }
51
52 return {
53 CATCH_MOVE(samples2),
54 wrap_estimate(analysis.mean),
55 wrap_estimate(analysis.standard_deviation),
56 outliers,
57 analysis.outlier_variance,
58 };
59 } else {
60 std::vector<FDuration> samples;
61 samples.reserve(static_cast<size_t>(last - first));
62
63 FDuration mean = FDuration(0);
64 int i = 0;
65 for (auto it = first; it < last; ++it, ++i) {
66 samples.push_back(FDuration(*it));
67 mean += FDuration(*it);
68 }
69 mean /= i;
70
71 return SampleAnalysis{
72 CATCH_MOVE(samples),
73 Estimate<FDuration>{ mean, mean, mean, 0.0 },
74 Estimate<FDuration>{ FDuration( 0 ),
75 FDuration( 0 ),
76 FDuration( 0 ),
77 0.0 },
78 OutlierClassification{},
79 0.0
80 };
81 }
82 }
