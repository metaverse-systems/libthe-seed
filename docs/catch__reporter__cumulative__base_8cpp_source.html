71 void
72 CumulativeReporterBase::sectionStarting( SectionInfo const& sectionInfo ) {
73 // We need a copy, because SectionStats expect to take ownership
74 SectionStats incompleteStats( SectionInfo(sectionInfo), Counts(), 0, false );
75 SectionNode* node;
76 if ( m_sectionStack.empty() ) {
77 if ( !m_rootSection ) {
78 m_rootSection =
79 Detail::make_unique<SectionNode>( incompleteStats );
80 }
81 node = m_rootSection.get();
82 } else {
83 SectionNode& parentNode = *m_sectionStack.back();
84 auto it = std::find_if( parentNode.childSections.begin(),
85 parentNode.childSections.end(),
86 BySectionInfo( sectionInfo ) );
87 if ( it == parentNode.childSections.end() ) {
88 auto newNode =
89 Detail::make_unique<SectionNode>( incompleteStats );
90 node = newNode.get();
91 parentNode.childSections.push_back( CATCH_MOVE( newNode ) );
92 } else {
93 node = it->get();
94 }
95 }
96
97 m_deepestSection = node;
98 m_sectionStack.push_back( node );
99 }

101 void CumulativeReporterBase::assertionEnded(
102 AssertionStats const& assertionStats ) {
103 assert( !m_sectionStack.empty() );
104 // AssertionResult holds a pointer to a temporary DecomposedExpression,
105 // which getExpandedExpression() calls to build the expression string.
106 // Our section stack copy of the assertionResult will likely outlive the
107 // temporary, so it must be expanded or discarded now to avoid calling
108 // a destroyed object later.
109 if ( m_shouldStoreFailedAssertions &&
110 !assertionStats.assertionResult.isOk() ) {
111 static_cast<void>(
112 assertionStats.assertionResult.getExpandedExpression() );
113 }
114 if ( m_shouldStoreSuccesfulAssertions &&
115 assertionStats.assertionResult.isOk() ) {
116 static_cast<void>(
117 assertionStats.assertionResult.getExpandedExpression() );
118 }
119 SectionNode& sectionNode = *m_sectionStack.back();
120 sectionNode.assertionsAndBenchmarks.emplace_back( assertionStats );
121 }

123 void CumulativeReporterBase::sectionEnded( SectionStats const& sectionStats ) {
124 assert( !m_sectionStack.empty() );
125 SectionNode& node = *m_sectionStack.back();
126 node.stats = sectionStats;

130 void CumulativeReporterBase::testCaseEnded(
131 TestCaseStats const& testCaseStats ) {
132 auto node = Detail::make_unique<TestCaseNode>( testCaseStats );
133 assert( m_sectionStack.size() == 0 );
134 node->children.push_back( CATCH_MOVE(m_rootSection) );
135 m_testCases.push_back( CATCH_MOVE(node) );
136
137 assert( m_deepestSection );
138 m_deepestSection->stdOut = testCaseStats.stdOut;
139 m_deepestSection->stdErr = testCaseStats.stdErr;
140 }
