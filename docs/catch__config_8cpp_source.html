37 static Optional<bazelShardingOptions> readBazelShardingOptions() {
38 const auto bazelShardIndex = Detail::getEnv( "TEST_SHARD_INDEX" );
39 const auto bazelShardTotal = Detail::getEnv( "TEST_TOTAL_SHARDS" );
40 const auto bazelShardInfoFile = Detail::getEnv( "TEST_SHARD_STATUS_FILE" );
41
42
43 const bool has_all =
44 bazelShardIndex && bazelShardTotal && bazelShardInfoFile;
45 if ( !has_all ) {
46 // We provide nice warning message if the input is
47 // misconfigured.
48 auto warn = []( const char* env_var ) {
49 Catch::cerr()
50 << "Warning: Bazel shard configuration is missing '"
51 << env_var << "'. Shard configuration is skipped.\n";
52 };
53 if ( !bazelShardIndex ) {
54 warn( "TEST_SHARD_INDEX" );
55 }
56 if ( !bazelShardTotal ) {
57 warn( "TEST_TOTAL_SHARDS" );
58 }
59 if ( !bazelShardInfoFile ) {
60 warn( "TEST_SHARD_STATUS_FILE" );
61 }
62 return {};
63 }
64
65 auto shardIndex = parseUInt( bazelShardIndex );
66 if ( !shardIndex ) {
67 Catch::cerr()
68 << "Warning: could not parse 'TEST_SHARD_INDEX' ('" << bazelShardIndex
69 << "') as unsigned int.\n";
70 return {};
71 }
72 auto shardTotal = parseUInt( bazelShardTotal );
73 if ( !shardTotal ) {
74 Catch::cerr()
75 << "Warning: could not parse 'TEST_TOTAL_SHARD' ('"
76 << bazelShardTotal << "') as unsigned int.\n";
77 return {};
78 }
79
80 return bazelShardingOptions{
81 *shardIndex, *shardTotal, bazelShardInfoFile };
82
83 }

95 Config::Config( ConfigData const& data ):
96 m_data( data ) {
97 // We need to trim filter specs to avoid trouble with superfluous
98 // whitespace (esp. important for bdd macros, as those are manually
99 // aligned with whitespace).
100
101 for (auto& elem : m_data.testsOrTags) {
102 elem = trim(elem);
103 }
104 for (auto& elem : m_data.sectionsToRun) {
105 elem = trim(elem);
106 }
107
108 // Insert the default reporter if user hasn't asked for a specific one
109 if ( m_data.reporterSpecifications.empty() ) {
110 m_data.reporterSpecifications.push_back( {
111#if defined( CATCH_CONFIG_DEFAULT_REPORTER )
112 CATCH_CONFIG_DEFAULT_REPORTER,
113#else
114 "console",
115#endif
116 {}, {}, {}
117 } );
118 }
119
120 if ( enableBazelEnvSupport() ) {
121 readBazelEnvVars();
122 }
123
124 // Bazel support can modify the test specs, so parsing has to happen
125 // after reading Bazel env vars.
126 TestSpecParser parser( ITagAliasRegistry::get() );
127 if ( !m_data.testsOrTags.empty() ) {
128 m_hasTestFilters = true;
129 for ( auto const& testOrTags : m_data.testsOrTags ) {
130 parser.parse( testOrTags );
131 }
132 }
133 m_testSpec = parser.testSpec();
134
135
136 // We now fixup the reporter specs to handle default output spec,
137 // default colour spec, etc
138 bool defaultOutputUsed = false;
139 for ( auto const& reporterSpec : m_data.reporterSpecifications ) {
140 // We do the default-output check separately, while always
141 // using the default output below to make the code simpler
142 // and avoid superfluous copies.
143 if ( reporterSpec.outputFile().none() ) {
144 CATCH_ENFORCE( !defaultOutputUsed,
145 "Internal error: cannot use default output for "
146 "multiple reporters" );
147 defaultOutputUsed = true;
148 }
149
150 m_processedReporterSpecs.push_back( ProcessedReporterSpec{
151 reporterSpec.name(),
152 reporterSpec.outputFile() ? *reporterSpec.outputFile()
153 : data.defaultOutputFilename,
154 reporterSpec.colourMode().valueOr( data.defaultColourMode ),
155 reporterSpec.customOptions() } );
156 }
157 }

214 void Config::readBazelEnvVars() {
215 // Register a JUnit reporter for Bazel. Bazel sets an environment
216 // variable with the path to XML output. If this file is written to
217 // during test, Bazel will not generate a default XML output.
218 // This allows the XML output file to contain higher level of detail
219 // than what is possible otherwise.
220 const auto bazelOutputFile = Detail::getEnv( "XML_OUTPUT_FILE" );
221
222 if ( bazelOutputFile ) {
223 m_data.reporterSpecifications.push_back(
224 { "junit", std::string( bazelOutputFile ), {}, {} } );
225 }
226
227 const auto bazelTestSpec = Detail::getEnv( "TESTBRIDGE_TEST_ONLY" );
228 if ( bazelTestSpec ) {
229 // Presumably the test spec from environment should overwrite
230 // the one we got from CLI (if we got any)
231 m_data.testsOrTags.clear();
232 m_data.testsOrTags.push_back( bazelTestSpec );
233 }
234
235 const auto bazelShardOptions = readBazelShardingOptions();
236 if ( bazelShardOptions ) {
237 std::ofstream f( bazelShardOptions->shardFilePath,
238 std::ios_base::out | std::ios_base::trunc );
239 if ( f.is_open() ) {
240 f << "";
241 m_data.shardIndex = bazelShardOptions->shardIndex;
242 m_data.shardCount = bazelShardOptions->shardCount;
243 }
244 }
245 }
