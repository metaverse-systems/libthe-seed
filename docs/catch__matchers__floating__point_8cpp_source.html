27 template <typename FP>
28 bool almostEqualUlps(FP lhs, FP rhs, uint64_t maxUlpDiff) {
29 // Comparison with NaN should always be false.
30 // This way we can rule it out before getting into the ugly details
31 if (Catch::isnan(lhs) || Catch::isnan(rhs)) {
32 return false;
33 }
34
35 // This should also handle positive and negative zeros, infinities
36 const auto ulpDist = ulpDistance(lhs, rhs);
37
38 return ulpDist <= maxUlpDiff;
39 }

53template <typename FP>
54FP step(FP start, FP direction, uint64_t steps) {
55 for (uint64_t i = 0; i < steps; ++i) {
56#if defined(CATCH_CONFIG_GLOBAL_NEXTAFTER)
57 start = Catch::nextafter(start, direction);
58#else
59 start = std::nextafter(start, direction);
60#endif
61 }
62 return start;
63}

65// Performs equivalent check of std::fabs(lhs - rhs) <= margin
66// But without the subtraction to allow for INFINITY in comparison
67bool marginComparison(double lhs, double rhs, double margin) {
68 return (lhs + margin >= rhs) && (rhs + margin >= lhs);
69}

110 CATCH_ENFORCE(m_type == Detail::FloatingPointKind::Double

124 bool WithinUlpsMatcher::match(double const& matchee) const {
125 switch (m_type) {
126 case Detail::FloatingPointKind::Float:
127 return almostEqualUlps<float>(static_cast<float>(matchee), static_cast<float>(m_target), m_ulps);
128 case Detail::FloatingPointKind::Double:
129 return almostEqualUlps<double>(matchee, m_target, m_ulps);
130 default:
131 CATCH_INTERNAL_ERROR( "Unknown Detail::FloatingPointKind value" );
132 }
133 }

139 std::string WithinUlpsMatcher::describe() const {
140 std::stringstream ret;
141
142 ret << "is within " << m_ulps << " ULPs of ";
143
144 if (m_type == Detail::FloatingPointKind::Float) {
145 write(ret, static_cast<float>(m_target));
146 ret << 'f';
147 } else {
148 write(ret, m_target);
149 }
150
151 ret << " ([";
152 if (m_type == Detail::FloatingPointKind::Double) {
153 write( ret,
154 step( m_target,
155 -std::numeric_limits<double>::infinity(),
156 m_ulps ) );
157 ret << ", ";
158 write( ret,
159 step( m_target,
160 std::numeric_limits<double>::infinity(),
161 m_ulps ) );
162 } else {
163 // We have to cast INFINITY to float because of MinGW, see #1782
164 write( ret,
165 step( static_cast<float>( m_target ),
166 -std::numeric_limits<float>::infinity(),
167 m_ulps ) );
168 ret << ", ";
169 write( ret,
170 step( static_cast<float>( m_target ),
171 std::numeric_limits<float>::infinity(),
172 m_ulps ) );
173 }
174 ret << "])";
175
176 return ret.str();
177 }

180 m_target(target),
181 m_epsilon(epsilon){
182 CATCH_ENFORCE(m_epsilon >= 0., "Relative comparison with epsilon < 0 does not make sense.");

186 bool WithinRelMatcher::match(double const& matchee) const {
187 const auto relMargin = m_epsilon * (std::max)(std::fabs(matchee), std::fabs(m_target));
188 return marginComparison(matchee, m_target,
189 std::isinf(relMargin)? 0 : relMargin);
190 }
