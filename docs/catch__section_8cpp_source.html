26 Section::Section( SourceLineInfo const& _lineInfo,
27 StringRef _name,
28 const char* const ):
29 m_info( { "invalid", static_cast<std::size_t>( -1 ) }, std::string{} ),
30 m_sectionIncluded(
31 getResultCapture().sectionStarted( _name, _lineInfo, m_assertions ) ) {
32 // We delay initialization the SectionInfo member until we know
33 // this section needs it, so we avoid allocating std::string for name.
34 // We also delay timer start to avoid the potential syscall unless we
35 // will actually use the result.
36 if ( m_sectionIncluded ) {
37 m_info.name = static_cast<std::string>( _name );
38 m_info.lineInfo = _lineInfo;
39 m_timer.start();
40 }
41 }

43 Section::~Section() {
44 if( m_sectionIncluded ) {
45 SectionEndInfo endInfo{ CATCH_MOVE(m_info), m_assertions, m_timer.getElapsedSeconds() };
46 if ( uncaught_exceptions() ) {
47 getResultCapture().sectionEndedEarly( CATCH_MOVE(endInfo) );
48 } else {
49 getResultCapture().sectionEnded( CATCH_MOVE( endInfo ) );
50 }
51 }
