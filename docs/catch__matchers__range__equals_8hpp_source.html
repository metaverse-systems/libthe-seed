24 template <typename TargetRangeLike, typename Equality>
25 class RangeEqualsMatcher final : public MatcherGenericBase {
26 TargetRangeLike m_desired;
27 Equality m_predicate;
28
29 public:
30 template <typename TargetRangeLike2, typename Equality2>
31 RangeEqualsMatcher( TargetRangeLike2&& range,
32 Equality2&& predicate ):
33 m_desired( CATCH_FORWARD( range ) ),
34 m_predicate( CATCH_FORWARD( predicate ) ) {}
35
36 template <typename RangeLike>
37 bool match( RangeLike&& rng ) const {
38 auto rng_start = begin( rng );
39 const auto rng_end = end( rng );
40 auto target_start = begin( m_desired );
41 const auto target_end = end( m_desired );
42
43 while (rng_start != rng_end && target_start != target_end) {
44 if (!m_predicate(*rng_start, *target_start)) {
45 return false;
46 }
47 ++rng_start;
48 ++target_start;
49 }
50 return rng_start == rng_end && target_start == target_end;
51 }
52
53 std::string describe() const override {
54 return "elements are " + Catch::Detail::stringify( m_desired );
55 }
56 };
