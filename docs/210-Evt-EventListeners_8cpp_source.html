314struct MyListener : Catch::EventListenerBase {
315
316 using EventListenerBase::EventListenerBase; // inherit constructor
317
318 // Get rid of Wweak-tables
319 ~MyListener() override;
320
321 // The whole test run starting
322 void testRunStarting( Catch::TestRunInfo const& testRunInfo ) override {
323 std::cout
324 << std::boolalpha
325 << "\nEvent: testRunStarting:\n";
326 print( std::cout, 1, "- testRunInfo", testRunInfo );
327 }
328
329 // The whole test run ending
330 void testRunEnded( Catch::TestRunStats const& testRunStats ) override {
331 std::cout
332 << dashed_line
333 << "\nEvent: testRunEnded:\n";
334 print( std::cout, 1, "- testRunStats", testRunStats );
335 }
336
337 // A test is being skipped (because it is "hidden")
338 void skipTest( Catch::TestCaseInfo const& testInfo ) override {
339 std::cout
340 << dashed_line
341 << "\nEvent: skipTest:\n";
342 print( std::cout, 1, "- testInfo", testInfo );
343 }
344
345 // Test cases starting
346 void testCaseStarting( Catch::TestCaseInfo const& testInfo ) override {
347 std::cout
348 << dashed_line
349 << "\nEvent: testCaseStarting:\n";
350 print( std::cout, 1, "- testInfo", testInfo );
351 }
352
353 // Test cases ending
354 void testCaseEnded( Catch::TestCaseStats const& testCaseStats ) override {
355 std::cout << "\nEvent: testCaseEnded:\n";
356 print( std::cout, 1, "testCaseStats", testCaseStats );
357 }
358
359 // Sections starting
360 void sectionStarting( Catch::SectionInfo const& sectionInfo ) override {
361 std::cout << "\nEvent: sectionStarting:\n";
362 print( std::cout, 1, "- sectionInfo", sectionInfo );
363 }
364
365 // Sections ending
366 void sectionEnded( Catch::SectionStats const& sectionStats ) override {
367 std::cout << "\nEvent: sectionEnded:\n";
368 print( std::cout, 1, "- sectionStats", sectionStats );
369 }
370
371 // Assertions before/ after
372 void assertionStarting( Catch::AssertionInfo const& assertionInfo ) override {
373 std::cout << "\nEvent: assertionStarting:\n";
374 print( std::cout, 1, "- assertionInfo", assertionInfo );
375 }
376
377 void assertionEnded( Catch::AssertionStats const& assertionStats ) override {
378 std::cout << "\nEvent: assertionEnded:\n";
379 print( std::cout, 1, "- assertionStats", assertionStats );
380 }
381};
