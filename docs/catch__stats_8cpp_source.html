31 template <typename URng, typename Estimator>
32 static sample
33 resample( URng& rng,
34 unsigned int resamples,
35 double const* first,
36 double const* last,
37 Estimator& estimator ) {
38 auto n = static_cast<size_t>( last - first );
39 std::uniform_int_distribution<decltype( n )> dist( 0,
40 n - 1 );
41
42 sample out;
43 out.reserve( resamples );
44 // We allocate the vector outside the loop to avoid realloc
45 // per resample
46 std::vector<double> resampled;
47 resampled.reserve( n );
48 for ( size_t i = 0; i < resamples; ++i ) {
49 resampled.clear();
50 for ( size_t s = 0; s < n; ++s ) {
51 resampled.push_back(
52 first[static_cast<std::ptrdiff_t>(
53 dist( rng ) )] );
54 }
55 const auto estimate =
56 estimator( resampled.data(), resampled.data() + resampled.size() );
57 out.push_back( estimate );
58 }
59 std::sort( out.begin(), out.end() );
60 return out;
61 }

63 static double outlier_variance( Estimate<double> mean,
64 Estimate<double> stddev,
65 int n ) {
66 double sb = stddev.point;
67 double mn = mean.point / n;
68 double mg_min = mn / 2.;
69 double sg = (std::min)( mg_min / 4., sb / std::sqrt( n ) );
70 double sg2 = sg * sg;
71 double sb2 = sb * sb;
72
73 auto c_max = [n, mn, sb2, sg2]( double x ) -> double {
74 double k = mn - x;
75 double d = k * k;
76 double nd = n * d;
77 double k0 = -n * nd;
78 double k1 = sb2 - n * sg2 + nd;
79 double det = k1 * k1 - 4 * sg2 * k0;
80 return static_cast<int>( -2. * k0 /
81 ( k1 + std::sqrt( det ) ) );
82 };
83
84 auto var_out = [n, sb2, sg2]( double c ) {
85 double nc = n - c;
86 return ( nc / n ) * ( sb2 - nc * sg2 );
87 };
88
89 return (std::min)( var_out( 1 ),
90 var_out(
91 (std::min)( c_max( 0. ),
92 c_max( mg_min ) ) ) ) /
93 sb2;
94 }

172 static double
173 standard_deviation( double const* first, double const* last ) {
174 auto m = Catch::Benchmark::Detail::mean( first, last );
175 double variance =
176 std::accumulate( first,
177 last,
178 0.,
179 [m]( double a, double b ) {
180 double diff = b - m;
181 return a + diff * diff;
182 } ) /
183 ( last - first );
184 return std::sqrt( variance );
185 }

201 static sample jackknife( double ( *estimator )( double const*,
202 double const* ),
203 double* first,
204 double* last ) {
205 const auto second = first + 1;
206 sample results;
207 results.reserve( static_cast<size_t>( last - first ) );
208
209 for ( auto it = first; it != last; ++it ) {
210 std::iter_swap( it, first );
211 results.push_back( estimator( second, last ) );
212 }
213
214 return results;
215 }

227 double weighted_average_quantile( int k,
228 int q,
229 double* first,
230 double* last ) {
231 auto count = last - first;
232 double idx = (count - 1) * k / static_cast<double>(q);
233 int j = static_cast<int>(idx);
234 double g = idx - j;
235 std::nth_element(first, first + j, last);
236 auto xj = first[j];
237 if ( directCompare( g, 0 ) ) {
238 return xj;
239 }
240
241 auto xj1 = *std::min_element(first + (j + 1), last);
242 return xj + g * (xj1 - xj);
243 }

245 OutlierClassification
246 classify_outliers( double const* first, double const* last ) {
247 std::vector<double> copy( first, last );
248
249 auto q1 = weighted_average_quantile( 1, 4, copy.data(), copy.data() + copy.size() );
250 auto q3 = weighted_average_quantile( 3, 4, copy.data(), copy.data() + copy.size() );
251 auto iqr = q3 - q1;
252 auto los = q1 - ( iqr * 3. );
253 auto lom = q1 - ( iqr * 1.5 );
254 auto him = q3 + ( iqr * 1.5 );
255 auto his = q3 + ( iqr * 3. );
256
257 OutlierClassification o;
258 for ( ; first != last; ++first ) {
259 const double t = *first;
260 if ( t < los ) {
261 ++o.low_severe;
262 } else if ( t < lom ) {
263 ++o.low_mild;
264 } else if ( t > his ) {
265 ++o.high_severe;
266 } else if ( t > him ) {
267 ++o.high_mild;
268 }
269 ++o.samples_seen;
270 }
271 return o;
272 }

274 double mean( double const* first, double const* last ) {
275 auto count = last - first;
276 double sum = 0.;
277 while (first != last) {
278 sum += *first;
279 ++first;
280 }
281 return sum / static_cast<double>(count);
282 }

292 double normal_quantile(double p) {
293 static const double ROOT_TWO = std::sqrt(2.0);
294
295 double result = 0.0;
296 assert(p >= 0 && p <= 1);
297 if (p < 0 || p > 1) {
298 return result;
299 }
300
301 result = -erfc_inv(2.0 * p);
302 // result *= normal distribution standard deviation (1.0) * sqrt(2)
303 result *= /*sd * */ ROOT_TWO;
304 // result += normal disttribution mean (0)
305 return result;
306 }

308 Estimate<double>
309 bootstrap( double confidence_level,
310 double* first,
311 double* last,
312 sample const& resample,
313 double ( *estimator )( double const*, double const* ) ) {
314 auto n_samples = last - first;
315
316 double point = estimator( first, last );
317 // Degenerate case with a single sample
318 if ( n_samples == 1 )
319 return { point, point, point, confidence_level };
320
321 sample jack = jackknife( estimator, first, last );
322 double jack_mean =
323 mean( jack.data(), jack.data() + jack.size() );
324 double sum_squares = 0, sum_cubes = 0;
325 for ( double x : jack ) {
326 auto difference = jack_mean - x;
327 auto square = difference * difference;
328 auto cube = square * difference;
329 sum_squares += square;
330 sum_cubes += cube;
331 }
332
333 double accel = sum_cubes / ( 6 * std::pow( sum_squares, 1.5 ) );
334 long n = static_cast<long>( resample.size() );
335 double prob_n =
336 std::count_if( resample.begin(),
337 resample.end(),
338 [point]( double x ) { return x < point; } ) /
339 static_cast<double>( n );
340 // degenerate case with uniform samples
341 if ( directCompare( prob_n, 0. ) ) {
342 return { point, point, point, confidence_level };
343 }
344
345 double bias = normal_quantile( prob_n );
346 double z1 = normal_quantile( ( 1. - confidence_level ) / 2. );
347
348 auto cumn = [n]( double x ) -> long {
349 return std::lround( normal_cdf( x ) *
350 static_cast<double>( n ) );
351 };
352 auto a = [bias, accel]( double b ) {
353 return bias + b / ( 1. - accel * b );
354 };
355 double b1 = bias + z1;
356 double b2 = bias - z1;
357 double a1 = a( b1 );
358 double a2 = a( b2 );
359 auto lo = static_cast<size_t>( (std::max)( cumn( a1 ), 0l ) );
360 auto hi =
361 static_cast<size_t>( (std::min)( cumn( a2 ), n - 1 ) );
362
363 return { point, resample[lo], resample[hi], confidence_level };
364 }

366 bootstrap_analysis analyse_samples(double confidence_level,
367 unsigned int n_resamples,
368 double* first,
369 double* last) {
370 CATCH_INTERNAL_START_WARNINGS_SUPPRESSION
371 CATCH_INTERNAL_SUPPRESS_GLOBALS_WARNINGS
372 static std::random_device entropy;
373 CATCH_INTERNAL_STOP_WARNINGS_SUPPRESSION
374
375 auto n = static_cast<int>(last - first); // seriously, one can't use integral types without hell in C++
376
377 auto mean = &Detail::mean;
378 auto stddev = &standard_deviation;
379
380#if defined(CATCH_CONFIG_USE_ASYNC)
381 auto Estimate = [=](double(*f)(double const*, double const*)) {
382 auto seed = entropy();
383 return std::async(std::launch::async, [=] {
384 std::mt19937 rng(seed);
385 auto resampled = resample(rng, n_resamples, first, last, f);
386 return bootstrap(confidence_level, first, last, resampled, f);
387 });
388 };
389
390 auto mean_future = Estimate(mean);
391 auto stddev_future = Estimate(stddev);
392
393 auto mean_estimate = mean_future.get();
394 auto stddev_estimate = stddev_future.get();
395#else
396 auto Estimate = [=](double(*f)(double const* , double const*)) {
397 auto seed = entropy();
398 std::mt19937 rng(seed);
399 auto resampled = resample(rng, n_resamples, first, last, f);
400 return bootstrap(confidence_level, first, last, resampled, f);
401 };
402
403 auto mean_estimate = Estimate(mean);
404 auto stddev_estimate = Estimate(stddev);
405#endif // CATCH_USE_ASYNC
406
407 double outlier_variance = Detail::outlier_variance(mean_estimate, stddev_estimate, n);
408
409 return { mean_estimate, stddev_estimate, outlier_variance };
410 }
