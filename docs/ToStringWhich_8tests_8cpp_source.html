110// Range-based conversion should only be used if other possibilities fail
111struct int_iterator {
112 using iterator_category = std::input_iterator_tag;
113 using difference_type = std::ptrdiff_t;
114 using value_type = int;
115 using reference = int&;
116 using pointer = int*;
117
118 int_iterator() = default;
119 int_iterator(int i) :val(i) {}
120
121 value_type operator*() const { return val; }
122 bool operator==(int_iterator rhs) const { return val == rhs.val; }
123 bool operator!=(int_iterator rhs) const { return val != rhs.val; }
124 int_iterator operator++() { ++val; return *this; }
125 int_iterator operator++(int) {
126 auto temp(*this);
127 ++val;
128 return temp;
129 }
130private:
131 int val = 5;
132};
