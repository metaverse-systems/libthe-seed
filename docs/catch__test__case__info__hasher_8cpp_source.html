15 uint32_t TestCaseInfoHasher::operator()( TestCaseInfo const& t ) const {
16 // FNV-1a hash algorithm that is designed for uniqueness:
17 const hash_t prime = 1099511628211u;

19 for ( const char c : t.name ) {
20 hash ^= c;
21 hash *= prime;
22 }
23 for ( const char c : t.className ) {
24 hash ^= c;
25 hash *= prime;
26 }

28 for ( const char c : tag.original ) {
29 hash ^= c;
30 hash *= prime;
31 }
32 }

34 hash *= prime;
35 const uint32_t low{ static_cast<uint32_t>( hash ) };
36 const uint32_t high{ static_cast<uint32_t>( hash >> 32 ) };
