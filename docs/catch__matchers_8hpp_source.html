47 template<typename ArgT>
48 class MatchAllOf final : public MatcherBase<ArgT> {
49 std::vector<MatcherBase<ArgT> const*> m_matchers;
50
51 public:
52 MatchAllOf() = default;
53 MatchAllOf(MatchAllOf const&) = delete;
54 MatchAllOf& operator=(MatchAllOf const&) = delete;
55 MatchAllOf(MatchAllOf&&) = default;
56 MatchAllOf& operator=(MatchAllOf&&) = default;
57
58
59 bool match( ArgT const& arg ) const override {
60 for( auto matcher : m_matchers ) {
61 if (!matcher->match(arg))
62 return false;
63 }
64 return true;
65 }
66 std::string describe() const override {
67 std::string description;
68 description.reserve( 4 + m_matchers.size()*32 );
69 description += "( ";
70 bool first = true;
71 for( auto matcher : m_matchers ) {
72 if( first )
73 first = false;
74 else
75 description += " and ";
76 description += matcher->toString();
77 }
78 description += " )";
79 return description;
80 }
81
82 friend MatchAllOf operator&& (MatchAllOf&& lhs, MatcherBase<ArgT> const& rhs) {
83 lhs.m_matchers.push_back(&rhs);
84 return CATCH_MOVE(lhs);
85 }
86 friend MatchAllOf operator&& (MatcherBase<ArgT> const& lhs, MatchAllOf&& rhs) {
87 rhs.m_matchers.insert(rhs.m_matchers.begin(), &lhs);
88 return CATCH_MOVE(rhs);
89 }
90 };
