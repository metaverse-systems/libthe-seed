28 template <typename Clock>
29 std::vector<double> resolution(int k) {
30 std::vector<TimePoint<Clock>> times;
31 times.reserve(static_cast<size_t>(k + 1));
32 for ( int i = 0; i < k + 1; ++i ) {
33 times.push_back( Clock::now() );
34 }
35
36 std::vector<double> deltas;
37 deltas.reserve(static_cast<size_t>(k));
38 for ( size_t idx = 1; idx < times.size(); ++idx ) {
39 deltas.push_back( static_cast<double>(
40 ( times[idx] - times[idx - 1] ).count() ) );
41 }
42
43 return deltas;
44 }

61 template <typename Clock>
62 EnvironmentEstimate estimate_clock_resolution(int iterations) {
63 auto r = run_for_at_least<Clock>(std::chrono::duration_cast<IDuration>(clock_resolution_estimation_time), iterations, &resolution<Clock>)
64 .result;
65 return {
66 FDuration(mean(r.data(), r.data() + r.size())),
67 classify_outliers(r.data(), r.data() + r.size()),
68 };

70 template <typename Clock>
71 EnvironmentEstimate estimate_clock_cost(FDuration resolution) {
72 auto time_limit = (std::min)(
73 resolution * clock_cost_estimation_tick_limit,
74 FDuration(clock_cost_estimation_time_limit));
75 auto time_clock = [](int k) {
76 return Detail::measure<Clock>([k] {
77 for (int i = 0; i < k; ++i) {
78 volatile auto ignored = Clock::now();
79 (void)ignored;
80 }
81 }).elapsed;
82 };
83 time_clock(1);
84 int iters = clock_cost_estimation_iterations;
85 auto&& r = run_for_at_least<Clock>(std::chrono::duration_cast<IDuration>(clock_cost_estimation_time), iters, time_clock);
86 std::vector<double> times;
87 int nsamples = static_cast<int>(std::ceil(time_limit / r.elapsed));
88 times.reserve(static_cast<size_t>(nsamples));
89 for ( int s = 0; s < nsamples; ++s ) {
90 times.push_back( static_cast<double>(
91 ( time_clock( r.iterations ) / r.iterations )
92 .count() ) );
93 }
94 return {
95 FDuration(mean(times.data(), times.data() + times.size())),
96 classify_outliers(times.data(), times.data() + times.size()),
97 };
98 }

101 Environment measure_environment() {
102#if defined(__clang__)
103# pragma clang diagnostic push
104# pragma clang diagnostic ignored "-Wexit-time-destructors"
105#endif
106 static Catch::Detail::unique_ptr<Environment> env;
107#if defined(__clang__)
108# pragma clang diagnostic pop
109#endif
110 if (env) {
111 return *env;
112 }
113
114 auto iters = Detail::warmup<Clock>();
115 auto resolution = Detail::estimate_clock_resolution<Clock>(iters);
116 auto cost = Detail::estimate_clock_cost<Clock>(resolution.mean);
117
118 env = Catch::Detail::make_unique<Environment>( Environment{resolution, cost} );
119 return *env;
120 }
