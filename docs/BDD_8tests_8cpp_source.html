47SCENARIO( "Vector resizing affects size and capacity",
48 "[vector][bdd][size][capacity]" ) {
49 GIVEN( "an empty vector" ) {
50 std::vector<int> v;
51 REQUIRE( v.size() == 0 );
52
53 WHEN( "it is made larger" ) {
54 v.resize( 10 );
55 THEN( "the size and capacity go up" ) {
56 REQUIRE( v.size() == 10 );
57 REQUIRE( v.capacity() >= 10 );
58
59 AND_WHEN( "it is made smaller again" ) {
60 v.resize( 5 );
61 THEN(
62 "the size goes down but the capacity stays the same" ) {
63 REQUIRE( v.size() == 5 );
64 REQUIRE( v.capacity() >= 10 );
65 }
66 }
67 }
68 }
69
70 WHEN( "we reserve more space" ) {
71 v.reserve( 10 );
72 THEN( "The capacity is increased but the size remains the same" ) {
73 REQUIRE( v.capacity() >= 10 );
74 REQUIRE( v.size() == 0 );
75 }
76 }
77 }
