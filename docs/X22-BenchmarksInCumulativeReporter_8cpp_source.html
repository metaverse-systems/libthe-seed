59void CumulativeBenchmarkReporter::testRunEndedCumulative() {
60 auto const& testCases = m_testRun->children;
61 assert(testCases.size() == 1);
62
63 auto const& testCase = *testCases.front();
64 auto const& sections = testCase.children;
65 assert(sections.size() == 1);
66
67 auto const& section = *sections.front();
68 assert(section.childSections.empty());
69 for (auto const& aob : section.assertionsAndBenchmarks) {
70 if (aob.isAssertion()) {
71 auto const& assertion = aob.asAssertion();
72 std::cout << assertion.assertionResult.getExpandedExpression() << '\n';
73 }
74 if (aob.isBenchmark()) {
