45 template <typename Clock, typename Fun>
46 TimingOf<Fun, run_for_at_least_argument_t<Clock, Fun>>
47 run_for_at_least(IDuration how_long,
48 const int initial_iterations,
49 Fun&& fun) {
50 auto iters = initial_iterations;
51 while (iters < (1 << 30)) {
52 auto&& Timing = measure_one<Clock>(fun, iters, is_callable<Fun(Chronometer)>());
53
54 if (Timing.elapsed >= how_long) {
55 return { Timing.elapsed, CATCH_MOVE(Timing.result), iters };
56 }
57 iters *= 2;
58 }
59 throw_optimized_away_error();
60 }
