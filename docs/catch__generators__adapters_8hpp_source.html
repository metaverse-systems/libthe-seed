20 template <typename T>
21 class TakeGenerator final : public IGenerator<T> {
22 GeneratorWrapper<T> m_generator;
23 size_t m_returned = 0;
24 size_t m_target;
25 public:
26 TakeGenerator(size_t target, GeneratorWrapper<T>&& generator):
27 m_generator(CATCH_MOVE(generator)),
28 m_target(target)
29 {
30 assert(target != 0 && "Empty generators are not allowed");
31 }
32 T const& get() const override {
33 return m_generator.get();
34 }
35 bool next() override {
36 ++m_returned;
37 if (m_returned >= m_target) {
38 return false;
39 }
40
41 const auto success = m_generator.next();
42 // If the underlying generator does not contain enough values
43 // then we cut short as well
44 if (!success) {
45 m_returned = m_target;
46 }
47 return success;
48 }
49 };

57 template <typename T, typename Predicate>
58 class FilterGenerator final : public IGenerator<T> {
59 GeneratorWrapper<T> m_generator;
60 Predicate m_predicate;
61 public:
62 template <typename P = Predicate>
63 FilterGenerator(P&& pred, GeneratorWrapper<T>&& generator):
64 m_generator(CATCH_MOVE(generator)),
65 m_predicate(CATCH_FORWARD(pred))
66 {
67 if (!m_predicate(m_generator.get())) {
68 // It might happen that there are no values that pass the
69 // filter. In that case we throw an exception.
70 auto has_initial_value = next();
71 if (!has_initial_value) {
72 Detail::throw_generator_exception("No valid value found in filtered generator");
73 }
74 }
75 }
76
77 T const& get() const override {
78 return m_generator.get();
79 }
80
81 bool next() override {
82 bool success = m_generator.next();
83 if (!success) {
84 return false;
85 }
86 while (!m_predicate(m_generator.get()) && (success = m_generator.next()) == true);
87 return success;
88 }
89 };

97 template <typename T>
98 class RepeatGenerator final : public IGenerator<T> {
99 static_assert(!std::is_same<T, bool>::value,
100 "RepeatGenerator currently does not support bools"
101 "because of std::vector<bool> specialization");
102 GeneratorWrapper<T> m_generator;
103 mutable std::vector<T> m_returned;
104 size_t m_target_repeats;
105 size_t m_current_repeat = 0;
106 size_t m_repeat_index = 0;
107 public:
108 RepeatGenerator(size_t repeats, GeneratorWrapper<T>&& generator):
109 m_generator(CATCH_MOVE(generator)),
110 m_target_repeats(repeats)
111 {
112 assert(m_target_repeats > 0 && "Repeat generator must repeat at least once");
113 }
114
115 T const& get() const override {
116 if (m_current_repeat == 0) {
117 m_returned.push_back(m_generator.get());
118 return m_returned.back();
119 }
120 return m_returned[m_repeat_index];
121 }
122
123 bool next() override {
124 // There are 2 basic cases:
125 // 1) We are still reading the generator
126 // 2) We are reading our own cache
127
128 // In the first case, we need to poke the underlying generator.
129 // If it happily moves, we are left in that state, otherwise it is time to start reading from our cache
130 if (m_current_repeat == 0) {
131 const auto success = m_generator.next();
132 if (!success) {
133 ++m_current_repeat;
134 }
135 return m_current_repeat < m_target_repeats;
136 }
137
138 // In the second case, we need to move indices forward and check that we haven't run up against the end
139 ++m_repeat_index;
140 if (m_repeat_index == m_returned.size()) {
141 m_repeat_index = 0;
142 ++m_current_repeat;
143 }
144 return m_current_repeat < m_target_repeats;
145 }
146 };

154 class MapGenerator final : public IGenerator<T> {
155 // TBD: provide static assert for mapping function, for friendly error message
156 GeneratorWrapper<U> m_generator;
157 Func m_function;
158 // To avoid returning dangling reference, we have to save the values
159 T m_cache;
160 public:
161 template <typename F2 = Func>
162 MapGenerator(F2&& function, GeneratorWrapper<U>&& generator) :
163 m_generator(CATCH_MOVE(generator)),
164 m_function(CATCH_FORWARD(function)),
165 m_cache(m_function(m_generator.get()))
166 {}
167
168 T const& get() const override {
169 return m_cache;
170 }
171 bool next() override {
172 const auto success = m_generator.next();
173 if (success) {
174 m_cache = m_function(m_generator.get());
175 }
176 return success;
177 }

194 template <typename T>
195 class ChunkGenerator final : public IGenerator<std::vector<T>> {
196 std::vector<T> m_chunk;
197 size_t m_chunk_size;
198 GeneratorWrapper<T> m_generator;
199 bool m_used_up = false;
200 public:
201 ChunkGenerator(size_t size, GeneratorWrapper<T> generator) :
202 m_chunk_size(size), m_generator(CATCH_MOVE(generator))
203 {
204 m_chunk.reserve(m_chunk_size);
205 if (m_chunk_size != 0) {
206 m_chunk.push_back(m_generator.get());
207 for (size_t i = 1; i < m_chunk_size; ++i) {
208 if (!m_generator.next()) {
209 Detail::throw_generator_exception("Not enough values to initialize the first chunk");
210 }
211 m_chunk.push_back(m_generator.get());
212 }
213 }
214 }
215 std::vector<T> const& get() const override {
216 return m_chunk;
217 }
218 bool next() override {
219 m_chunk.clear();
220 for (size_t idx = 0; idx < m_chunk_size; ++idx) {
221 if (!m_generator.next()) {
222 return false;
223 }
224 m_chunk.push_back(m_generator.get());
225 }
226 return true;
227 }
228 };
