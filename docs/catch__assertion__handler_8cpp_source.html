41 void AssertionHandler::complete() {
42 m_completed = true;
43 if( m_reaction.shouldDebugBreak ) {
44
45 // If you find your debugger stopping you here then go one level up on the
46 // call-stack for the code that caused it (typically a failed assertion)
47
48 // (To go back to the test and change execution, jump over the throw, next)
49 CATCH_BREAK_INTO_DEBUGGER();
50 }
51 if (m_reaction.shouldThrow) {
52 throw_test_failure_exception();
53 }
54 if ( m_reaction.shouldSkip ) {
55 throw_test_skip_exception();
56 }
57 }
