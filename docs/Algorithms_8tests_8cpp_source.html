17 template <typename Range1, typename Range2>
18 static bool is_permutation(Range1 const& r1, Range2 const& r2) {
19 using std::begin; using std::end;
20 return Catch::Detail::is_permutation(
21 begin( r1 ), end( r1 ), begin( r2 ), end( r2 ), std::equal_to<>{} );
22 }
23}

25TEST_CASE("is_permutation", "[algorithms][approvals]") {
26 SECTION( "Handle empty ranges" ) {
27 std::array<int, 0> empty;
28 std::array<int, 2> non_empty{ { 2, 3 } };
29 REQUIRE( is_permutation( empty, empty ) );
30 REQUIRE_FALSE( is_permutation( empty, non_empty ) );
31 REQUIRE_FALSE( is_permutation( non_empty, empty ) );
32 }
33 SECTION( "Different length ranges" ) {
34 std::array<int, 6> arr1{ { 1, 3, 5, 7, 8, 9 } };
35 // arr2 is prefix of arr1
36 std::array<int, 4> arr2{ { 1, 3, 5, 7 } };
37 // arr3 shares prefix with arr1 and arr2, but is not a permutation
38 std::array<int, 5> arr3{ { 1, 3, 5, 9, 8 } };
39 REQUIRE_FALSE( is_permutation( arr1, arr2 ) );
40 REQUIRE_FALSE( is_permutation( arr1, arr3 ) );
41 REQUIRE_FALSE( is_permutation( arr2, arr3 ) );
42 }
43 SECTION( "Same length ranges" ) {
44 SECTION( "Shared elements, but different counts" ) {
45 const std::array<int, 6>
46 arr1{ { 1, 1, 1, 1, 2, 2 } },
47 arr2{ { 1, 1, 2, 2, 2, 2 } };
48 REQUIRE_FALSE( is_permutation( arr1, arr2 ) );
