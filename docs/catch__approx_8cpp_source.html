17// Performs equivalent check of std::fabs(lhs - rhs) <= margin
18// But without the subtraction to allow for INFINITY in comparison
19bool marginComparison(double lhs, double rhs, double margin) {
20 return (lhs + margin >= rhs) && (rhs + margin >= lhs);
21}

51 bool Approx::equalityComparisonImpl(const double other) const {
52 // First try with fixed margin, then compute margin based on epsilon, scale and Approx's value
53 // Thanks to Richard Harris for his help refining the scaled margin value
54 return marginComparison(m_value, other, m_margin)
55 || marginComparison(m_value, other, m_epsilon * (m_scale + std::fabs(std::isinf(m_value)? 0 : m_value)));
56 }
