167TEST_CASE( "Exception messages can be tested for", "[!throws]" ) {
168 using namespace Catch::Matchers;
169 SECTION( "exact match" )
170 REQUIRE_THROWS_WITH( thisThrows(), "expected exception" );
171 SECTION( "different case" )
172 REQUIRE_THROWS_WITH( thisThrows(), Equals( "expecteD Exception", Catch::CaseSensitive::No ) );
173 SECTION( "wildcarded" ) {
174 REQUIRE_THROWS_WITH( thisThrows(), StartsWith( "expected" ) );
175 REQUIRE_THROWS_WITH( thisThrows(), EndsWith( "exception" ) );
176 REQUIRE_THROWS_WITH( thisThrows(), ContainsSubstring( "except" ) );
