65 // Different type for the "end" iterator
66 struct iterator_end {};
67 // Fake-ish forward iterator that only compares to a different type
68 class iterator {

70 underlying_iter m_start;

78 using difference_type = std::ptrdiff_t;
79 using value_type = T;

121template <typename T>
122struct with_mocked_iterator_access {
123 std::vector<T> m_elements;
124
125 // use plain arrays to have nicer printouts with CHECK(...)
126 mutable std::unique_ptr<bool[]> m_derefed;
127
128 // We want to check which elements were dereferenced when iterating, so
129 // we can check whether iterator-using code traverses range correctly
130 template <bool is_const>
131 class basic_iterator {
132 template <typename U>
133 using constify_t = std::conditional_t<is_const, std::add_const_t<U>, U>;
134
135 constify_t<with_mocked_iterator_access>* m_origin;
136 size_t m_origin_idx;
137
138 public:
139 using iterator_category = std::forward_iterator_tag;
140 using difference_type = std::ptrdiff_t;
141 using value_type = constify_t<T>;
142 using const_reference = typename std::vector<T>::const_reference;
143 using reference = typename std::vector<T>::reference;
144 using pointer = typename std::vector<T>::pointer;
145
146 basic_iterator( constify_t<with_mocked_iterator_access>* origin,
147 std::size_t origin_idx ):
148 m_origin{ origin }, m_origin_idx{ origin_idx } {}
149
150 friend bool operator==( basic_iterator lhs, basic_iterator rhs ) {
151 return lhs.m_origin == rhs.m_origin &&
152 lhs.m_origin_idx == rhs.m_origin_idx;
153 }
154 friend bool operator!=( basic_iterator lhs, basic_iterator rhs ) {
155 return !( lhs == rhs );
156 }
157 basic_iterator& operator++() {
158 ++m_origin_idx;
159 return *this;
160 }
161 basic_iterator operator++( int ) {
162 auto tmp( *this );
163 ++( *this );
164 return tmp;
165 }
166 const_reference operator*() const {
167 assert( m_origin_idx < m_origin->m_elements.size() &&
168 "Attempted to deref invalid position" );
169 m_origin->m_derefed[m_origin_idx] = true;
170 return m_origin->m_elements[m_origin_idx];
171 }
172 pointer operator->() const {
173 assert( m_origin_idx < m_origin->m_elements.size() &&
174 "Attempted to deref invalid position" );
175 return &m_origin->m_elements[m_origin_idx];
176 }
177 };
178
179 using iterator = basic_iterator<false>;
180 using const_iterator = basic_iterator<true>;
181
182 with_mocked_iterator_access( std::initializer_list<T> init ):
183 m_elements( init ),
184 m_derefed( std::make_unique<bool[]>( m_elements.size() ) ) {}
185
186 const_iterator begin() const { return { this, 0 }; }
187 const_iterator end() const { return { this, m_elements.size() }; }
188 iterator begin() { return { this, 0 }; }
189 iterator end() { return { this, m_elements.size() }; }
190};
