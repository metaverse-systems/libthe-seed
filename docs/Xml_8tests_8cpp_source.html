17static std::string encode( std::string const& str, Catch::XmlEncode::ForWhat forWhat = Catch::XmlEncode::ForTextNodes ) {
18 Catch::ReusableStringStream oss;
19 oss << Catch::XmlEncode( str, forWhat );

23TEST_CASE( "XmlEncode", "[XML]" ) {
24 SECTION( "normal string" ) {
25 REQUIRE( encode( "normal string" ) == "normal string" );
26 }
27 SECTION( "empty string" ) {
28 REQUIRE( encode( "" ) == "" );
29 }
30 SECTION( "string with ampersand" ) {
31 REQUIRE( encode( "smith & jones" ) == "smith &amp; jones" );
32 }
33 SECTION( "string with less-than" ) {
34 REQUIRE( encode( "smith < jones" ) == "smith &lt; jones" );
35 }
36 SECTION( "string with greater-than" ) {
37 REQUIRE( encode( "smith > jones" ) == "smith > jones" );
38 REQUIRE( encode( "smith ]]> jones" ) == "smith ]]&gt; jones" );
39 }
40 SECTION( "string with quotes" ) {
41 std::string stringWithQuotes = "don't \"quote\" me on that";
42 REQUIRE( encode( stringWithQuotes ) == stringWithQuotes );
43 REQUIRE( encode( stringWithQuotes, Catch::XmlEncode::ForAttributes ) == "don't &quot;quote&quot; me on that" );
44 }
45 SECTION( "string with control char (1)" ) {
46 REQUIRE( encode( "[\x01]" ) == "[\\x01]" );
47 }
48 SECTION( "string with control char (x7F)" ) {
49 REQUIRE( encode( "[\x7F]" ) == "[\\x7F]" );
50 }
51}

54TEST_CASE("XmlEncode: UTF-8", "[XML][UTF-8][approvals]") {
55 SECTION("Valid utf-8 strings") {
56 CHECK(encode("Here be 游") == "Here be 游");
57 CHECK(encode("코코") == "코코");
58
59 CHECK(encode("\xDF\xBF") == "\xDF\xBF"); // 0x7FF
60 CHECK(encode("\xE0\xA0\x80") == "\xE0\xA0\x80"); // 0x800
61 CHECK(encode("\xED\x9F\xBF") == "\xED\x9F\xBF"); // 0xD7FF
62 CHECK(encode("\xEE\x80\x80") == "\xEE\x80\x80"); // 0xE000
63 CHECK(encode("\xEF\xBF\xBF") == "\xEF\xBF\xBF"); // 0xFFFF
64 CHECK(encode("\xF0\x90\x80\x80") == "\xF0\x90\x80\x80"); // 0x10000
65 CHECK(encode("\xF4\x8F\xBF\xBF") == "\xF4\x8F\xBF\xBF"); // 0x10FFFF
66 }
67 SECTION("Invalid utf-8 strings") {
68 SECTION("Various broken strings") {
69 CHECK(encode("Here \xFF be \xF0\x9F\x91\xBE") == "Here \\xFF be 游");
70 CHECK(encode("\xFF") == "\\xFF");
71 CHECK(encode("\xC5\xC5\xA0") == "\\xC5");
72 CHECK(encode("\xF4\x90\x80\x80") == "\\xF4\\x90\\x80\\x80"); // 0x110000 -- out of unicode range
73 }
74
75 SECTION("Overlong encodings") {
76 CHECK(encode("\xC0\x80") == "\\xC0\\x80"); // \0
77 CHECK(encode("\xF0\x80\x80\x80") == "\\xF0\\x80\\x80\\x80"); // Super-over-long \0
78 CHECK(encode("\xC1\xBF") == "\\xC1\\xBF"); // ASCII char as UTF-8 (0x7F)
79 CHECK(encode("\xE0\x9F\xBF") == "\\xE0\\x9F\\xBF"); // 0x7FF
80 CHECK(encode("\xF0\x8F\xBF\xBF") == "\\xF0\\x8F\\xBF\\xBF"); // 0xFFFF
81 }
