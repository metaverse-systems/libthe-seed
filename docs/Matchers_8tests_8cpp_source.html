229template <typename T> struct CustomAllocator : private std::allocator<T> {
230 using size_type = size_t;
231 using difference_type = ptrdiff_t;
232 using pointer = T*;
233 using const_pointer = const T*;
234 using reference = T&;
235 using const_reference = const T&;
236 using value_type = T;
237
238 template <typename U> struct rebind { using other = CustomAllocator<U>; };
239
240 using propagate_on_container_move_assignment = std::true_type;
241 using is_always_equal = std::true_type;
242
243 CustomAllocator() = default;
244
245 CustomAllocator( const CustomAllocator& other ):
246 std::allocator<T>( other ) {}
247
248 template <typename U> CustomAllocator( const CustomAllocator<U>& ) {}
249
250 ~CustomAllocator() = default;
251
252 using std::allocator<T>::allocate;
253 using std::allocator<T>::deallocate;
254};

274 v4.push_back( 3 + 1e-8 );

276 std::vector<int, CustomAllocator<int>> v5;
277 v5.push_back( 1 );
278 v5.push_back( 2 );
279 v5.push_back( 3 );
280

282 v6.push_back( 1 );
283 v6.push_back( 2 );

287 SECTION( "Contains (element)" ) {
288 CHECK_THAT( v, VectorContains( 1 ) );

295 CHECK_THAT( v5,

297 v2 ) ) );
298
299 v2.push_back( 3 ); // now exactly matches
300 CHECK_THAT( v, Contains( v2 ) );
301
302 CHECK_THAT( v, Contains( empty ) );
303 CHECK_THAT( empty, Contains( empty ) );
304
305 CHECK_THAT( v5,
306 ( Contains<int, std::allocator<int>, CustomAllocator<int>>(
307 v2 ) ) );
308 CHECK_THAT( v5, Contains( v6 ) );
309 }
310 SECTION( "Contains (element), composed" ) {
311 CHECK_THAT( v, VectorContains( 1 ) && VectorContains( 2 ) );
312 }

314 SECTION( "Equals" ) {
315
316 // Same vector
317 CHECK_THAT( v, Equals( v ) );
318
319 CHECK_THAT( empty, Equals( empty ) );
320

324 CHECK_THAT( v, Equals( v2 ) );
325
326 CHECK_THAT(
327 v5,
328 ( Equals<int, std::allocator<int>, CustomAllocator<int>>( v2 ) ) );
