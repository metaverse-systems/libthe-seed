23 class ExceptionTranslatorRegistrar {
24 template<typename T>
25 class ExceptionTranslator : public IExceptionTranslator {
26 public:
27
28 ExceptionTranslator( std::string(*translateFunction)( T const& ) )
29 : m_translateFunction( translateFunction )
30 {}
31
32 std::string translate( ExceptionTranslators::const_iterator it, ExceptionTranslators::const_iterator itEnd ) const override {
33#if !defined(CATCH_CONFIG_DISABLE_EXCEPTIONS)
34 try {
35 if( it == itEnd )
36 std::rethrow_exception(std::current_exception());
37 else
38 return (*it)->translate( it+1, itEnd );
39 }
40 catch( T const& ex ) {
41 return m_translateFunction( ex );
42 }
43#else
44 return "You should never get here!";
45#endif
46 }
47
48 protected:
49 std::string(*m_translateFunction)( T const& );
50 };
51
52 public:
53 template<typename T>
54 ExceptionTranslatorRegistrar( std::string(*translateFunction)( T const& ) ) {
55 Detail::registerTranslatorImpl(
56 Detail::make_unique<ExceptionTranslator<T>>(
57 translateFunction ) );
58 }
59 };
