22BinaryFormat DetectFormat(const std::string &binary_path)
23{
24 std::ifstream input(binary_path, std::ios::binary);
25 if(!input.is_open())
26 {
27 throw std::runtime_error("Unable to open file");
28 }
29
30 std::uint8_t header[4] = {};
31 input.read(reinterpret_cast<char *>(header), sizeof(header));
32 if(input.gcount() < 2)
33 {
34 throw std::runtime_error("File is too small to determine binary format");
35 }
36
37 if(input.gcount() >= 4 && header[0] == ELFMAG0 && header[1] == ELFMAG1 && header[2] == ELFMAG2 &&
38 header[3] == ELFMAG3)
39 {
40 return BinaryFormat::Elf;
41 }
42
43 if(header[0] == 'M' && header[1] == 'Z')
44 {
45 return BinaryFormat::Pe;
46 }
47
48 throw std::runtime_error("Unsupported binary format");
49}
