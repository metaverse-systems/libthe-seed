132TEST_CASE("Precision of floating point stringification can be set", "[toString][floatingPoint]") {
133 SECTION("Floats") {
134 using sm = Catch::StringMaker<float>;
135 const auto oldPrecision = sm::precision;
136
137 const float testFloat = 1.12345678901234567899f;
138 auto str1 = sm::convert(testFloat);
139 sm::precision = 5;
140 // "1." prefix = 2 chars, f suffix is another char
141 CHECK(str1.size() == 3 + 5);
142
143 sm::precision = 10;
144 auto str2 = sm::convert(testFloat);
145 REQUIRE(str2.size() == 3 + 10);
146 sm::precision = oldPrecision;
147 }
148 SECTION("Double") {
149 using sm = Catch::StringMaker<double>;
150 const auto oldPrecision = sm::precision;
151
152 const double testDouble = 1.123456789012345678901234567899;
153 sm::precision = 5;

155 // "1." prefix = 2 chars

157
158 sm::precision = 15;
