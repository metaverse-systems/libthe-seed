54 template <typename FP>
55 uint64_t ulpDistance( FP lhs, FP rhs ) {
56 assert( std::numeric_limits<FP>::is_iec559 &&
57 "ulpDistance assumes IEEE-754 format for floating point types" );
58 assert( !Catch::isnan( lhs ) &&
59 "Distance between NaN and number is not meaningful" );
60 assert( !Catch::isnan( rhs ) &&
61 "Distance between NaN and number is not meaningful" );
62
63 // We want X == Y to imply 0 ULP distance even if X and Y aren't
64 // bit-equal (-0 and 0), or X - Y != 0 (same sign infinities).
65 if ( lhs == rhs ) { return 0; }
66
67 // We need a properly typed positive zero for type inference.
68 static constexpr FP positive_zero{};
69
70 // We want to ensure that +/- 0 is always represented as positive zero
71 if ( lhs == positive_zero ) { lhs = positive_zero; }
72 if ( rhs == positive_zero ) { rhs = positive_zero; }
73
74 // If arguments have different signs, we can handle them by summing
75 // how far are they from 0 each.
76 if ( std::signbit( lhs ) != std::signbit( rhs ) ) {
77 return ulpDistance( std::abs( lhs ), positive_zero ) +
78 ulpDistance( std::abs( rhs ), positive_zero );
79 }
80
81 // When both lhs and rhs are of the same sign, we can just
82 // read the numbers bitwise as integers, and then subtract them
83 // (assuming IEEE).
84 uint64_t lc = Detail::convertToBits( lhs );
85 uint64_t rc = Detail::convertToBits( rhs );
86
87 // The ulp distance between two numbers is symmetric, so to avoid
88 // dealing with overflows we want the bigger converted number on the lhs
89 if ( lc < rc ) {
90 std::swap( lc, rc );
91 }
92
93 return lc - rc;
94 }

97# pragma GCC diagnostic pop
