34namespace {
35 /* Minimal Allocator */
36 template<typename T>
37 struct minimal_allocator {
38 using value_type = T;
39 using size_type = std::size_t;
40
41 minimal_allocator() = default;
42 template <typename U>
43 minimal_allocator(const minimal_allocator<U>&) {}
44
45
46 T *allocate( size_type n ) {
47 return static_cast<T *>( ::operator new( n * sizeof(T) ) );
48 }
49 void deallocate( T *p, size_type /*n*/ ) {
50 ::operator delete( static_cast<void *>(p) );
51 }
52 template<typename U>
53 bool operator==( const minimal_allocator<U>& ) const { return true; }
54 template<typename U>
55 bool operator!=( const minimal_allocator<U>& ) const { return false; }
56 };
57}
