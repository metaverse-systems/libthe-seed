54TEST_CASE("Clara::Opt supports accept-many lambdas", "[clara][opt]") {
55 using namespace Catch::Clara;
56 std::vector<std::string> res;
57 const auto push_to_res = [&](std::string const& s) {
58 res.push_back(s);
59 return ParserResult::ok( ParseResultType::Matched );
60 };
61
62 SECTION("Parsing fails on multiple options without accept_many") {
63 auto p = Parser() | Opt(push_to_res, "value")["-o"];
64 auto parse_result = p.parse( Args{ "UnitTest", "-o", "aaa", "-o", "bbb" } );
65 CHECK_FALSE(parse_result);
66 }
67 SECTION("Parsing succeeds on multiple options with accept_many") {
68 auto p = Parser() | Opt(accept_many, push_to_res, "value")["-o"];
69 auto parse_result = p.parse( Args{ "UnitTest", "-o", "aaa", "-o", "bbb" } );
70 CHECK(parse_result);
71 CHECK(res == std::vector<std::string>{ "aaa", "bbb" });
72 }
73}
