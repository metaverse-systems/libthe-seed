9// 231-Cfg-OutputStreams.cpp
10// Show how to replace the streams with a simple custom made streambuf.
11
12// Note that this reimplementation _does not_ follow `std::cerr`
13// semantic, because it buffers the output. For most uses however,
14// there is no important difference between having `std::cerr` buffered
15// or unbuffered.
16#include <catch2/catch_test_macros.hpp>
17
18#include <sstream>
19#include <cstdio>
20
21class out_buff : public std::stringbuf {
22 std::FILE* m_stream;
23public:
24 out_buff(std::FILE* stream):m_stream(stream) {}
25 ~out_buff();
26 int sync() override {
27 int ret = 0;
28 for (unsigned char c : str()) {
29 if (putc(c, m_stream) == EOF) {
30 ret = -1;
31 break;
32 }
33 }
34 // Reset the buffer to avoid printing it multiple times
35 str("");
36 return ret;
37 }
38};
39
40out_buff::~out_buff() { pubsync(); }
