18TEST_CASE("Trim strings", "[string-manip]") {
19 using Catch::trim; using Catch::StringRef;
20 static_assert(std::is_same<std::string, decltype(trim(std::string{}))>::value, "Trimming std::string should return std::string");
21 static_assert(std::is_same<StringRef, decltype(trim(StringRef{}))>::value, "Trimming StringRef should return StringRef");
22
23 REQUIRE(trim(std::string(no_whitespace)) == no_whitespace);
24 REQUIRE(trim(std::string(leading_whitespace)) == no_whitespace);
25 REQUIRE(trim(std::string(trailing_whitespace)) == no_whitespace);
26 REQUIRE(trim(std::string(whitespace_at_both_ends)) == no_whitespace);
27
28 REQUIRE(trim(StringRef(no_whitespace)) == StringRef(no_whitespace));
29 REQUIRE(trim(StringRef(leading_whitespace)) == StringRef(no_whitespace));
30 REQUIRE(trim(StringRef(trailing_whitespace)) == StringRef(no_whitespace));
31 REQUIRE(trim(StringRef(whitespace_at_both_ends)) == StringRef(no_whitespace));
32}

34TEST_CASE("replaceInPlace", "[string-manip]") {
35 std::string letters = "abcdefcg";
36 SECTION("replace single char") {
37 CHECK(Catch::replaceInPlace(letters, "b", "z"));

40 SECTION("replace two chars") {

42 CHECK(letters == "abzdefzg");

44 SECTION("replace first char") {
45 CHECK(Catch::replaceInPlace(letters, "a", "z"));
46 CHECK(letters == "zbcdefcg");
47 }
48 SECTION("replace last char") {
49 CHECK(Catch::replaceInPlace(letters, "g", "z"));
50 CHECK(letters == "abcdefcz");
51 }
52 SECTION("replace all chars") {
53 CHECK(Catch::replaceInPlace(letters, letters, "replaced"));
54 CHECK(letters == "replaced");
55 }
56 SECTION("replace no chars") {
57 CHECK_FALSE(Catch::replaceInPlace(letters, "x", "z"));
