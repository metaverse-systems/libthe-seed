9// 311-Gen-CustomCapture.cpp
10// Shows how to provide custom capture list to the generator expression
11
12// Note that using variables inside generators is dangerous and should
13// be done only if you know what you are doing, because the generators
14// _WILL_ outlive the variables. Also, even if you know what you are
15// doing, you should probably use GENERATE_COPY or GENERATE_REF macros
16// instead. However, if your use case requires having a
17// per-variable custom capture list, this example shows how to achieve
18// that.
19
20#include <catch2/catch_test_macros.hpp>
21#include <catch2/generators/catch_generators_adapters.hpp>
22#include <catch2/generators/catch_generators_random.hpp>
23
24TEST_CASE("Generate random doubles across different ranges",
25 "[generator][example][advanced]") {
26 // Workaround for old libstdc++
27 using record = std::tuple<double, double>;
28 // Set up 3 ranges to generate numbers from
29 auto r1 = GENERATE(table<double, double>({
30 record{3, 4},
31 record{-4, -3},
32 record{10, 1000}
33 }));
34
35 auto r2(r1);
36
37 // This will take r1 by reference and r2 by value.
38 // Note that there are no advantages for doing so in this example,
39 // it is done only for expository purposes.
40 auto number = Catch::Generators::generate( "custom capture generator", CATCH_INTERNAL_LINEINFO,
41 [&r1, r2]{
42 using namespace Catch::Generators;
43 return makeGenerators(take(50, random(std::get<0>(r1), std::get<1>(r2))));
44 }
45 );
46
47 REQUIRE(std::abs(number) > 0);
48}
