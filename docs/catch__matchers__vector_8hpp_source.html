42 template<typename T, typename AllocComp, typename AllocMatch>
43 class ContainsMatcher final : public MatcherBase<std::vector<T, AllocMatch>> {
44 std::vector<T, AllocComp> const& m_comparator;
45
46 public:
47 ContainsMatcher(std::vector<T, AllocComp> const& comparator):
48 m_comparator( comparator )
49 {}
50
51 bool match(std::vector<T, AllocMatch> const& v) const override {
52 // !TBD: see note in EqualsMatcher
53 if (m_comparator.size() > v.size())
54 return false;
55 for (auto const& comparator : m_comparator) {
56 auto present = false;
57 for (const auto& el : v) {
58 if (el == comparator) {
59 present = true;
60 break;
61 }
62 }
63 if (!present) {
64 return false;
65 }
66 }
67 return true;
68 }
69 std::string describe() const override {
70 return "Contains: " + ::Catch::Detail::stringify( m_comparator );
71 }

74 template<typename T, typename AllocComp, typename AllocMatch>
75 class EqualsMatcher final : public MatcherBase<std::vector<T, AllocMatch>> {
76 std::vector<T, AllocComp> const& m_comparator;
77
78 public:
79 EqualsMatcher(std::vector<T, AllocComp> const& comparator):
80 m_comparator( comparator )
81 {}
82
83 bool match(std::vector<T, AllocMatch> const& v) const override {
84 // !TBD: This currently works if all elements can be compared using !=
85 // - a more general approach would be via a compare template that defaults
86 // to using !=. but could be specialised for, e.g. std::vector<T> etc
87 // - then just call that directly
88 if ( m_comparator.size() != v.size() ) { return false; }
89 for ( std::size_t i = 0; i < v.size(); ++i ) {
90 if ( !( m_comparator[i] == v[i] ) ) { return false; }
91 }
92 return true;
93 }
94 std::string describe() const override {
95 return "Equals: " + ::Catch::Detail::stringify( m_comparator );
96 }

99 template<typename T, typename AllocComp, typename AllocMatch>
100 class ApproxMatcher final : public MatcherBase<std::vector<T, AllocMatch>> {
101 std::vector<T, AllocComp> const& m_comparator;
102 mutable Catch::Approx approx = Catch::Approx::custom();
103
104 public:
105 ApproxMatcher(std::vector<T, AllocComp> const& comparator):
106 m_comparator( comparator )
107 {}
108
109 bool match(std::vector<T, AllocMatch> const& v) const override {
110 if (m_comparator.size() != v.size())
111 return false;
112 for (std::size_t i = 0; i < v.size(); ++i)
113 if (m_comparator[i] != approx(v[i]))
114 return false;
115 return true;
116 }
117 std::string describe() const override {
118 return "is approx: " + ::Catch::Detail::stringify( m_comparator );
119 }

137 template<typename T, typename AllocComp, typename AllocMatch>
138 class UnorderedEqualsMatcher final : public MatcherBase<std::vector<T, AllocMatch>> {
139 std::vector<T, AllocComp> const& m_target;
140
141 public:
142 UnorderedEqualsMatcher(std::vector<T, AllocComp> const& target):
143 m_target(target)
144 {}
145 bool match(std::vector<T, AllocMatch> const& vec) const override {
146 if (m_target.size() != vec.size()) {
147 return false;
148 }
149 return std::is_permutation(m_target.begin(), m_target.end(), vec.begin());
150 }
151
152 std::string describe() const override {
153 return "UnorderedEquals: " + ::Catch::Detail::stringify(m_target);
154 }
155 };
