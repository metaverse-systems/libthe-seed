55 // Given IEEE-754 format for floats and doubles, we can assume
56 // that float -> double promotion is lossless. Given this, we can
57 // assume that if we do the standard relative comparison of
58 // |lhs - rhs| <= epsilon * max(fabs(lhs), fabs(rhs)), then we get
59 // the same result if we do this for floats, as if we do this for
60 // doubles that were promoted from floats.
61 class WithinRelMatcher final : public MatcherBase<double> {
62 public:
63 WithinRelMatcher( double target, double epsilon );
64 bool match(double const& matchee) const override;
65 std::string describe() const override;
66 private:
67 double m_target;
68 double m_epsilon;
69 };
