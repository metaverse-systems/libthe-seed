20template <typename T>
21class RangeGenerator final : public IGenerator<T> {
22 T m_current;
23 T m_end;
24 T m_step;
25 bool m_positive;
26
27public:
28 RangeGenerator(T const& start, T const& end, T const& step):
29 m_current(start),
30 m_end(end),
31 m_step(step),
32 m_positive(m_step > T(0))
33 {
34 assert(m_current != m_end && "Range start and end cannot be equal");
35 assert(m_step != T(0) && "Step size cannot be zero");
36 assert(((m_positive && m_current <= m_end) || (!m_positive && m_current >= m_end)) && "Step moves away from end");
37 }
38
39 RangeGenerator(T const& start, T const& end):
40 RangeGenerator(start, end, (start < end) ? T(1) : T(-1))
41 {}
42
43 T const& get() const override {
44 return m_current;
45 }
46
47 bool next() override {
48 m_current += m_step;
49 return (m_positive) ? (m_current < m_end) : (m_current > m_end);
50 }
51};

66template <typename T>
67class IteratorGenerator final : public IGenerator<T> {
68 static_assert(!std::is_same<T, bool>::value,
69 "IteratorGenerator currently does not support bools"
70 "because of std::vector<bool> specialization");
71
72 std::vector<T> m_elems;
73 size_t m_current = 0;
74public:
75 template <typename InputIterator, typename InputSentinel>
76 IteratorGenerator(InputIterator first, InputSentinel last):m_elems(first, last) {
77 if (m_elems.empty()) {
78 Detail::throw_generator_exception("IteratorGenerator received no valid values");
79 }
80 }
81
82 T const& get() const override {
83 return m_elems[m_current];
84 }
85
86 bool next() override {
87 ++m_current;
88 return m_current != m_elems.size();
89 }
90};
