81
82 bool RegexMatcher::match(std::string const& matchee) const {
83 auto flags = std::regex::ECMAScript; // ECMAScript is the default syntax option anyway
84 if (m_caseSensitivity == CaseSensitive::No) {
85 flags |= std::regex::icase;
86 }
87 auto reg = std::regex(m_regex, flags);
88 return std::regex_match(matchee, reg);
89 }
90
91 std::string RegexMatcher::describe() const {
92 return "matches " + ::Catch::Detail::stringify(m_regex) + ((m_caseSensitivity == CaseSensitive::Yes)? " case sensitively" : " case insensitively");
93 }
