26template <typename Float>
27class RandomFloatingGenerator final : public IGenerator<Float> {
28 Catch::SimplePcg32 m_rng;
29 std::uniform_real_distribution<Float> m_dist;
30 Float m_current_number;
31public:
32 RandomFloatingGenerator( Float a, Float b, std::uint32_t seed ):
33 m_rng(seed),
34 m_dist(a, b) {
35 static_cast<void>(next());
36 }
37
38 Float const& get() const override {
39 return m_current_number;
40 }
41 bool next() override {
42 m_current_number = m_dist(m_rng);
43 return true;
44 }
45};

68template <typename T>
69std::enable_if_t<std::is_integral<T>::value, GeneratorWrapper<T>>
70random(T a, T b) {
71 static_assert(
72 !std::is_same<T, char>::value &&
73 !std::is_same<T, int8_t>::value &&
74 !std::is_same<T, uint8_t>::value &&
75 !std::is_same<T, signed char>::value &&
76 !std::is_same<T, unsigned char>::value &&
77 !std::is_same<T, bool>::value,
78 "The requested type is not supported by the underlying random distributions from std" );
79 return GeneratorWrapper<T>(
80 Catch::Detail::make_unique<RandomIntegerGenerator<T>>(a, b, Detail::getSeed())
81 );
82}
