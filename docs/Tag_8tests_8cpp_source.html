15TEST_CASE( "Tag alias can be registered against tag patterns" ) {
16
17 Catch::TagAliasRegistry registry;
18
19 registry.add( "[@zzz]", "[one][two]", Catch::SourceLineInfo( "file", 2 ) );
20
21 SECTION( "The same tag alias can only be registered once" ) {
22
23 try {
24 registry.add( "[@zzz]", "[one][two]", Catch::SourceLineInfo( "file", 10 ) );
25 FAIL( "expected exception" );
26 }
27 catch( std::exception& ex ) {
28 std::string what = ex.what();
29 using namespace Catch::Matchers;
30 CHECK_THAT( what, ContainsSubstring( "[@zzz]" ) );
31 CHECK_THAT( what, ContainsSubstring( "file" ) );
32 CHECK_THAT( what, ContainsSubstring( "2" ) );
33 CHECK_THAT( what, ContainsSubstring( "10" ) );
34 }
35 }
36
37 SECTION( "Tag aliases must be of the form [@name]" ) {
38 CHECK_THROWS( registry.add( "[no ampersat]", "", Catch::SourceLineInfo( "file", 3 ) ) );
39 CHECK_THROWS( registry.add( "[the @ is not at the start]", "", Catch::SourceLineInfo( "file", 3 ) ) );
40 CHECK_THROWS( registry.add( "@no square bracket at start]", "", Catch::SourceLineInfo( "file", 3 ) ) );
41 CHECK_THROWS( registry.add( "[@no square bracket at end", "", Catch::SourceLineInfo( "file", 3 ) ) );
42 }
43}
