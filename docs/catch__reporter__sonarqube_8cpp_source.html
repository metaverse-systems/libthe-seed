63 void SonarQubeReporter::writeTestCase(TestCaseNode const& testCaseNode) {
64 // All test cases have exactly one section - which represents the

66 assert(testCaseNode.children.size() == 1);

71 void SonarQubeReporter::writeSection(std::string const& rootName, SectionNode const& sectionNode, bool okToFail) {
72 std::string name = trim(sectionNode.stats.sectionInfo.name);
73 if (!rootName.empty())
74 name = rootName + '/' + name;
75
76 if ( sectionNode.hasAnyAssertions()
77 || !sectionNode.stdOut.empty()
78 || !sectionNode.stdErr.empty() ) {
79 XmlWriter::ScopedElement e = xml.scopedElement("testCase");
80 xml.writeAttribute("name"_sr, name);
81 xml.writeAttribute("duration"_sr, static_cast<long>(sectionNode.stats.durationInSeconds * 1000));
82
83 writeAssertions(sectionNode, okToFail);
84 }
85
86 for (auto const& childNode : sectionNode.childSections)
87 writeSection(name, *childNode, okToFail);
88 }

98 void SonarQubeReporter::writeAssertion(AssertionStats const& stats, bool okToFail) {
99 AssertionResult const& result = stats.assertionResult;
100 if ( !result.isOk() ||
101 result.getResultType() == ResultWas::ExplicitSkip ) {
102 std::string elementName;
103 if (okToFail) {
104 elementName = "skipped";
105 } else {
106 switch (result.getResultType()) {
107 case ResultWas::ThrewException:
108 case ResultWas::FatalErrorCondition:
109 elementName = "error";
110 break;
111 case ResultWas::ExplicitFailure:
112 case ResultWas::ExpressionFailed:
113 case ResultWas::DidntThrowException:
114 elementName = "failure";
115 break;
116 case ResultWas::ExplicitSkip:
117 elementName = "skipped";
118 break;
119 // We should never see these here:
120 case ResultWas::Info:
121 case ResultWas::Warning:
122 case ResultWas::Ok:
123 case ResultWas::Unknown:
124 case ResultWas::FailureBit:
125 case ResultWas::Exception:
126 elementName = "internalError";
127 break;
128 }
129 }
130
131 XmlWriter::ScopedElement e = xml.scopedElement(elementName);
132
133 ReusableStringStream messageRss;
134 messageRss << result.getTestMacroName() << '(' << result.getExpression() << ')';
135 xml.writeAttribute("message"_sr, messageRss.str());
136
137 ReusableStringStream textRss;
138 if ( result.getResultType() == ResultWas::ExplicitSkip ) {
139 textRss << "SKIPPED\n";
140 } else {
141 textRss << "FAILED:\n";
142 if (result.hasExpression()) {
143 textRss << '\t' << result.getExpressionInMacro() << '\n';
144 }
145 if (result.hasExpandedExpression()) {
146 textRss << "with expansion:\n\t" << result.getExpandedExpression() << '\n';
147 }
148 }
149
150 if (result.hasMessage())
151 textRss << result.getMessage() << '\n';
152
153 for (auto const& msg : stats.infoMessages)
154 if (msg.type == ResultWas::Info)
155 textRss << msg.message << '\n';
156
157 textRss << "at " << result.getSourceInfo();
158 xml.writeText(textRss.str(), XmlFormatting::Newline);
159 }
160 }
