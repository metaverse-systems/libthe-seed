68TEST_CASE( "variant<variant<float, int>, string>", "[toString][variant][approvals]")
69{
70 using inner = std::variant<MyType1, float, int>;
71 using type = std::variant<inner, std::string>;
72 CHECK( "0.5f" == ::Catch::Detail::stringify(type{0.5f}) );
73 CHECK( "0" == ::Catch::Detail::stringify(type{0}) );
74 CHECK( "\"foo\"" == ::Catch::Detail::stringify(type{"foo"}) );
75
76 SECTION("valueless nested variant") {
77 type value = inner{0.5f};
78 REQUIRE( std::holds_alternative<inner>(value) );
79 REQUIRE( std::holds_alternative<float>(std::get<inner>(value)) );
80
81 REQUIRE_THROWS_AS( std::get<0>(value).emplace<MyType1>(MyType1{}), int );
82
83 // outer variant is still valid and contains inner
84 REQUIRE( std::holds_alternative<inner>(value) );
85 // inner variant is valueless
86 REQUIRE( std::get<inner>(value).valueless_by_exception() );
87 CHECK( "{valueless variant}" == ::Catch::Detail::stringify(value) );
88 }
89}
