14TEST_CASE("Stringifying std::chrono::duration helpers", "[toString][chrono]") {
15 // No literals because we still support c++11
16 auto hour = std::chrono::hours(1);
17 auto minute = std::chrono::minutes(1);
18 auto seconds = std::chrono::seconds(60);
19 auto micro = std::chrono::microseconds(1);
20 auto milli = std::chrono::milliseconds(1);
21 auto nano = std::chrono::nanoseconds(1);
22 REQUIRE(minute == seconds);
23 REQUIRE(hour != seconds);
24 REQUIRE(micro != milli);
25 REQUIRE(nano != micro);
26}
