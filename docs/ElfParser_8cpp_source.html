32std::vector<std::uint8_t> ReadFileBytes(const std::string &file_path)
33{
34 std::ifstream input(file_path, std::ios::binary);
35 if(!input.is_open())
36 {
37 throw std::runtime_error("Unable to open file");
38 }
39
40 input.seekg(0, std::ios::end);

47 }
48

50 if(size > 0)

52 input.read(reinterpret_cast<char *>(bytes.data()), size);

55 throw std::runtime_error("Unable to read file");

78std::string ReadCString(const std::vector<std::uint8_t> &bytes, std::size_t offset)
79{
80 if(offset >= bytes.size())
81 {
82 throw std::runtime_error("Invalid string table offset in ELF");
83 }
84
85 std::size_t end = offset;
86 while(end < bytes.size() && bytes[end] != 0)
87 {
88 ++end;
89 }
90
91 if(end == bytes.size())
92 {
93 throw std::runtime_error("Unterminated dependency string in ELF");
94 }
95
96 return std::string(
97 reinterpret_cast<const char *>(bytes.data() + offset),
98 reinterpret_cast<const char *>(bytes.data() + end)

102std::vector<std::string> ParseElf32(const std::vector<std::uint8_t> &bytes, bool file_is_little_endian)
103{
104 const auto header = ReadStruct<Elf32_Ehdr>(bytes, 0);
105
106 const std::uint64_t phoff = ByteSwapIfNeeded(header.e_phoff, file_is_little_endian);
107 const std::uint16_t phentsize = ByteSwapIfNeeded(header.e_phentsize, file_is_little_endian);
108 const std::uint16_t phnum = ByteSwapIfNeeded(header.e_phnum, file_is_little_endian);
109
110 if(phentsize < sizeof(Elf32_Phdr))
111 {
112 throw std::runtime_error("Invalid ELF program header entry size");
113 }
114
115 std::optional<std::uint64_t> dynamic_offset;
116 std::uint64_t dynamic_size = 0;
117 std::vector<LoadSegment> load_segments;
118
119 for(std::uint16_t index = 0; index < phnum; ++index)
120 {
121 const std::uint64_t entry_offset = phoff + static_cast<std::uint64_t>(index) * phentsize;
122 const auto phdr = ReadStruct<Elf32_Phdr>(bytes, static_cast<std::size_t>(entry_offset));
123
124 const std::uint32_t p_type = ByteSwapIfNeeded(phdr.p_type, file_is_little_endian);
125 const std::uint32_t p_offset = ByteSwapIfNeeded(phdr.p_offset, file_is_little_endian);
126 const std::uint32_t p_vaddr = ByteSwapIfNeeded(phdr.p_vaddr, file_is_little_endian);
127 const std::uint32_t p_memsz = ByteSwapIfNeeded(phdr.p_memsz, file_is_little_endian);
128 const std::uint32_t p_filesz = ByteSwapIfNeeded(phdr.p_filesz, file_is_little_endian);
129
130 if(p_type == PT_LOAD)
131 {
132 load_segments.push_back({p_vaddr, p_memsz, p_offset});
133 }
134
135 if(p_type == PT_DYNAMIC)
136 {
137 dynamic_offset = p_offset;
138 dynamic_size = p_filesz;
139 }
140 }
141
142 if(!dynamic_offset.has_value() || dynamic_size == 0)
143 {
144 return {};
145 }
146
147 std::vector<std::uint64_t> needed_offsets;
148 std::optional<std::uint64_t> string_table_va;
149
150 for(std::uint64_t offset = *dynamic_offset; offset + sizeof(Elf32_Dyn) <= *dynamic_offset + dynamic_size;
151 offset += sizeof(Elf32_Dyn))
152 {
153 const auto dyn = ReadStruct<Elf32_Dyn>(bytes, static_cast<std::size_t>(offset));
154 const std::int32_t tag = ByteSwapIfNeeded(dyn.d_tag, file_is_little_endian);
155
156 if(tag == DT_NULL)
157 {
158 break;
159 }
160
161 if(tag == DT_NEEDED)
162 {
163 needed_offsets.push_back(ByteSwapIfNeeded(dyn.d_un.d_val, file_is_little_endian));
164 }
165 else if(tag == DT_STRTAB)
166 {
167 string_table_va = ByteSwapIfNeeded(dyn.d_un.d_ptr, file_is_little_endian);
168 }
169 }
170
171 if(!string_table_va.has_value())
172 {
173 throw std::runtime_error("Missing DT_STRTAB in ELF dynamic section");
174 }
175
176 const auto string_table_offset = VirtualAddressToOffset(*string_table_va, load_segments);
177 if(!string_table_offset.has_value())
178 {
179 throw std::runtime_error("Unable to resolve ELF string table offset");
180 }
181
182 std::vector<std::string> dependencies;
183 dependencies.reserve(needed_offsets.size());
184
185 for(const auto needed_offset : needed_offsets)
186 {
187 dependencies.push_back(ReadCString(bytes, static_cast<std::size_t>(*string_table_offset + needed_offset)));
188 }
189
190 return dependencies;
191}

285std::vector<std::string> ElfParser::ListDependencies(const std::string &file_path)
286{
287 const auto bytes = ReadFileBytes(file_path);
288 if(bytes.size() < EI_NIDENT)
289 {
290 throw std::runtime_error("File is too small to be an ELF binary");
291 }
292
293 if(bytes[EI_MAG0] != ELFMAG0 || bytes[EI_MAG1] != ELFMAG1 || bytes[EI_MAG2] != ELFMAG2 ||
294 bytes[EI_MAG3] != ELFMAG3)
295 {
296 throw std::runtime_error("Invalid ELF magic");
297 }
298
299 const std::uint8_t elf_class = bytes[EI_CLASS];
300 const std::uint8_t elf_data = bytes[EI_DATA];
301
302 if(elf_data != ELFDATA2LSB && elf_data != ELFDATA2MSB)
303 {
304 throw std::runtime_error("Unsupported ELF endianness");
305 }
306
307 const bool file_is_little_endian = elf_data == ELFDATA2LSB;
308
309 if(elf_class == ELFCLASS32)
310 {
311 return ParseElf32(bytes, file_is_little_endian);
312 }
313
314 if(elf_class == ELFCLASS64)
315 {
316 return ParseElf64(bytes, file_is_little_endian);
317 }
318
319 throw std::runtime_error("Unsupported ELF class");
320}
