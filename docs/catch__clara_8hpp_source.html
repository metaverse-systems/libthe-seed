249 assert( m_type != ResultType::RuntimeError );
250 if ( m_type != ResultType::Ok )

291 template <typename T>
292 ParserResult convertInto( std::string const& source, T& target ) {
293 std::stringstream ss( source );
294 ss >> target;
295 if ( ss.fail() ) {
296 return ParserResult::runtimeError(

301 }

370 template <typename ReturnType> struct LambdaInvoker {
371 static_assert(
372 std::is_same<ReturnType, ParserResult>::value,
373 "Lambda must return void or clara::ParserResult" );
374
375 template <typename L, typename ArgType>
376 static auto invoke( L const& lambda, ArgType const& arg )
377 -> ParserResult {
378 return lambda( arg );
379 }
380 };
381
382 template <> struct LambdaInvoker<void> {
383 template <typename L, typename ArgType>
384 static auto invoke( L const& lambda, ArgType const& arg )
385 -> ParserResult {
386 lambda( arg );
387 return ParserResult::ok( ParseResultType::Matched );
388 }
389 };
390
391 template <typename ArgType, typename L>
392 auto invokeLambda( L const& lambda, std::string const& arg )
393 -> ParserResult {
394 ArgType temp{};
395 auto result = convertInto( arg, temp );
396 return !result ? result
397 : LambdaInvoker<typename UnaryLambdaTraits<
398 L>::ReturnType>::invoke( lambda, temp );
399 }
