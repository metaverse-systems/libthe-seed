25TEMPLATE_TEST_CASE("type-shared ulpDistance tests", "[floating-point][ulp][approvals]", float, double) {
26 using FP = TestType;
27 using Catch::ulpDistance;
28
29 // Distance between zeros is zero
30 CHECK( ulpDistance( FP{}, FP{} ) == 0 );
31 CHECK( ulpDistance( FP{}, -FP{} ) == 0 );
32 CHECK( ulpDistance( -FP{}, -FP{} ) == 0 );
33
34 // Distance between same-sign infinities is zero
35 static constexpr FP infinity = std::numeric_limits<FP>::infinity();
36 CHECK( ulpDistance( infinity, infinity ) == 0 );
37 CHECK( ulpDistance( -infinity, -infinity ) == 0 );
38
39 // Distance between max-finite-val and same sign infinity is 1
40 static constexpr FP max_finite = std::numeric_limits<FP>::max();
41 CHECK( ulpDistance( max_finite, infinity ) == 1 );
42 CHECK( ulpDistance( -max_finite, -infinity ) == 1 );
43
44 // Distance between X and 0 is half of distance between X and -X
45 CHECK( ulpDistance( -infinity, infinity ) ==
46 2 * ulpDistance( infinity, FP{} ) );
47 CHECK( 2 * ulpDistance( FP{ -2. }, FP{} ) ==
48 ulpDistance( FP{ -2. }, FP{ 2. } ) );
49 CHECK( 2 * ulpDistance( FP{ 2. }, FP{} ) ==
50 ulpDistance( FP{ -2. }, FP{ 2. } ) );
51
52 // Denorms are supported
53 CHECK( ulpDistance( std::numeric_limits<FP>::denorm_min(), FP{} ) == 1 );
54 CHECK( ulpDistance( std::numeric_limits<FP>::denorm_min(), -FP{} ) == 1 );
55 CHECK( ulpDistance( -std::numeric_limits<FP>::denorm_min(), FP{} ) == 1 );
56 CHECK( ulpDistance( -std::numeric_limits<FP>::denorm_min(), -FP{} ) == 1 );
57 CHECK( ulpDistance( std::numeric_limits<FP>::denorm_min(),
58 -std::numeric_limits<FP>::denorm_min() ) == 2 );
59
60 // Machine epsilon
61 CHECK( ulpDistance( FP{ 1. },
62 FP{ 1. } + std::numeric_limits<FP>::epsilon() ) == 1 );
63 CHECK( ulpDistance( -FP{ 1. },
64 -FP{ 1. } - std::numeric_limits<FP>::epsilon() ) == 1 );
65}
