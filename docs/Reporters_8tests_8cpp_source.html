194TEST_CASE("Multireporter calls reporters and listeners in correct order",
195 "[reporters][multi-reporter]") {
196 Catch::Config config( Catch::ConfigData{} );
197
198 // We add reporters before listeners, to check that internally they
199 // get sorted properly, and listeners are called first anyway.
200 Catch::MultiReporter multiReporter( &config );
201 std::vector<std::string> records;
202 multiReporter.addReporter( Catch::Detail::make_unique<MockReporter>(
203 "Goodbye", records, makeDummyRepConfig(config) ) );
204 multiReporter.addListener(
205 Catch::Detail::make_unique<MockListener>( "Hello", records, &config ) );
206 multiReporter.addListener(
207 Catch::Detail::make_unique<MockListener>( "world", records, &config ) );
208 multiReporter.addReporter( Catch::Detail::make_unique<MockReporter>(
209 "world", records, makeDummyRepConfig(config) ) );
210 multiReporter.testRunStarting( { "" } );
211
212 std::vector<std::string> expected( { "Hello", "world", "Goodbye", "world" } );
213 REQUIRE( records == expected );
214}
