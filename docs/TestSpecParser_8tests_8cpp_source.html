18 static Catch::TestSpec parseAndCreateSpec(std::string const& str) {
19 Catch::TagAliasRegistry registry;
20 Catch::TestSpecParser parser( registry );
21
22 parser.parse( str );
23 auto spec = parser.testSpec();
24 REQUIRE( spec.hasFilters() );
25 REQUIRE( spec.getInvalidSpecs().empty());
26
27 return spec;
28 }
