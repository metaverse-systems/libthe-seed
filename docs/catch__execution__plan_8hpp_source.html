24 struct ExecutionPlan {

31 template <typename Clock>
32 std::vector<FDuration> run(const IConfig &cfg, Environment env) const {
33 // warmup a bit
34 Detail::run_for_at_least<Clock>(
35 std::chrono::duration_cast<IDuration>( warmup_time ),
36 warmup_iterations,
37 Detail::repeat( []() { return Clock::now(); } )
38 );
39
40 std::vector<FDuration> times;
41 const auto num_samples = cfg.benchmarkSamples();
42 times.reserve( num_samples );
43 for ( size_t i = 0; i < num_samples; ++i ) {
44 Detail::ChronometerModel<Clock> model;
45 this->benchmark( Chronometer( model, iterations_per_sample ) );
46 auto sample_time = model.elapsed() - env.clock_cost.mean;
47 if ( sample_time < FDuration::zero() ) {
48 sample_time = FDuration::zero();
49 }
50 times.push_back(sample_time / iterations_per_sample);
51 }
52 return times;
53 }
54 };
